### ការវិនិយោគសម្រាប់ថ្ងៃសប្បាតហ៍ទី១៣

ស្ត្រីជនជាតិអូស្ត្រាលីម្នាក់បានពិនិត្យមើលឡើងវិញ ទៅលើគម្រោងបេសកកម្មនៃថ្ងៃសប្បាតហ៍ ទី១៣ នៅពេលដែលក្រុមជំនុំរបស់នាងបានលើកទឹកចិត្តសមាជិក ឱ្យថ្វាយលើសពីអ្វីដែលពួកគេមាន នៅក្នុងហោប៉ៅរបស់ពួកគេ ជារៀងរាល់ថ្ងៃសប្បាតហ៍ចុងក្រោយប្រចាំត្រីមាស។

នាងម៉ារីឡេន ស្ទេវេនសុន (Marilene Stevenson) បានចងចាំពេលវេលាមួយ នៅពេលដែលសមាជិកក្រុមជំនុំបានដុតនំ និងដាំបន្លែ ហើយយករបស់ទាំងនេះ ទៅលក់យកប្រាក់សម្រាប់គម្រោង វិនិយោគនៃថ្នាក់សប្បាតហ៍ស្គូល។ នាងម៉ារីឡេនបានគិតថា នាងប្រហែលជាអាចធ្វើអ្វីមួយស្រដៀងគ្នា នេះ ដោយការផ្តល់ថវិកាជួយដល់តង្វាយនៃថ្ងៃសប្បាតហ៍ទី១៣ ដែលគេបានស្គាល់ថា ជាតង្វាយនៃ គម្រោងបេសកកម្មប្រចាំត្រីមាស។

នាងម៉ារីឡេនមិនមានពេលវេលា ឬមានគម្រោងនៅក្នុងការដុតនំ និងដាំបន្លែទេ។ ប៉ុន្តែ បើទោះ ជាយ៉ាងនេះក្តី នាងបានចាប់ផ្តើមប្រើថ្នាំបាញ់បំពង់ក ដែលនាងបានរកឃើញថា មានប្រយោជន៍ ហើយ បានណែនាំថ្នាំនោះដល់អ្នកដទៃ។ នៅពេលដែលអ្នកដទៃទៀតបានចាប់អារម្មណ៍លើថ្នាំបាញ់ ដែលនាង បានណែនាំ នោះនាងបាននឹកឃើញពីរបស់ដុំ យកមកលក់រាយវិញ។ ប្រាក់ចំណេញទាំងអស់ នឹងត្រូវ ប្រគល់ទៅឱ្យតង្វាយនៃថ្ងៃសប្បាតហ៍ទី១៣។

ក្នុងការវិនិយោគថ្នាំបាញ់បំពង់កជាលើកដំបូង បានជំរុញឱ្យនាងម៉ារីឡេនស្វែងរកមធ្យោបាយ បន្ថែមទៀត ដើម្បីរកប្រាក់សម្រាប់បេសកកម្មនេះ។ នាងបាននិយាយថា «ការនេះបានចាប់ផ្តើម ជាមួយ នឹងការដែលខ្ញុំទិញរបស់ ដែលខ្ញុំចង់បាន ប៉ុន្តែ របស់ដែលខ្ញុំចូលចិត្តទិញនោះ បានក្លាយជារបស់ដែល ចង់ទិញដែរ គ្រាន់តែថា ពួកគេមិនមានពេលវេលា ទៅទិញរបស់ទាំងនេះតែប៉ុណ្ណោះ»។

ដោយបានធ្វើការជាមួយកសិដ្ឋាន និងរោងចក្រក្នុងស្រុក នាងម៉ារីឡេនអាចទិញផលិតផល ដែលមានសុខភាពល្អទៅតាមតម្លៃលក់ដុំ ហើយយករបស់ទាំងនោះ មកលក់ឱ្យគ្រួសារ និងមិត្តភក្តិ ថោក ជាងតម្លៃលក់រាយ។ នាងបានលក់ផ្លែឈើដែលកម្រមាន តៅហ៊ូសាច់ស និងអាហារវត្ថុធាតុដើមសម្រាប់ អ្នកបួស អស់រយៈពេលជាច្រើនឆ្នាំ។ នាងក៏បានធ្វើការរៀបចំផ្កា ហើយលក់ផ្កាទាំងនោះ សម្រាប់ឱកាស ពិសេសដូចជាទិវារមាតាជាដើម។ នាងបាននិយាយថា «បេសកកម្មទាំងនេះបានធ្វើឱ្យខ្ញុំមានអារម្មណ៍ ពិតប្រាកដមួយ ពោលគឺខ្ញុំកំពុងតែជួយដល់អ្នក ដែលខ្ញុំបានលក់ឱ្យ ប៉ុន្តែ ទន្ទឹមគ្នានេះ ខ្ញុំក៏បានរកប្រាក់ សម្រាប់បេសកកម្មមួយយ៉ាងល្អផងដែរ»។

ចាប់តាំងពីនាងបានចាប់ផ្តើមវិនិយោគនេះមក នាងម៉ារីឡេនបានរកប្រាក់ពី២០០ទៅ៤០០ដុល្លារជារៀងរាល់ត្រីមាស ដើម្បីជួយដល់គម្រោងនៃបេសកកម្មថ្ងៃសប្បាតហ៍ទី១៣។ នាងបានលើក ទឹកចិត្តសមាជិកក្រុមជំនុំទាំងអស់ ឱ្យជួយផ្តល់បរិច្ចាគថវិកា ជួយដល់គម្រោងនៃសប្បាតហ៍ទី១៣ ទោះបី ជាការបរិច្ចាគនេះ បានទាមទារឱ្យមានការវិនិយោគ ដើម្បីឱ្យបានជួយបន្ថែមទៀតក៏ដោយ។

នាងម៉ារីឡេនបាននិយាយថា «ខ្ញុំសង្ឃឹមថា រឿងនេះនឹងបំផុសគំនិតអ្នកដទៃទៀត ឱ្យមាន ឱកាសធ្វើអ្វីមួយ ឱ្យលើសពីការថ្វាយតង្វាយបន្តិចបន្តួច។ ព្រះអម្ចាស់បានប្រទានពរដល់ខ្ញុំពិតប្រាកដមែន ដោយសារតែបទពិសោធន៍ដំបូង នៃភាពស្មោះត្រង់ដ៏អស្ចារ្យរបស់ព្រះអង្គ។ ដូច្នេះហើយ ខ្ញុំត្រូវតែ ចែករំលែកបទពិសោធន៍នេះ ដល់បងប្អូនយើង»។
