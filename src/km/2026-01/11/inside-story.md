### ប្រឈមមុខនឹងការសម្រេចចិត្តចុងក្រោយ

នៅថ្ងៃសប្បាតហ៍ដែលលោករេនបានសម្រេចចិត្តទៅព្រះវិហារ ជំនួសឱ្យការទៅការិយាល័យ នោះគាត់បានត្រឡប់មកវិញ ទៅអគារដែលគាត់រស់នៅ និងធ្វើការ បន្ទាប់ពីថ្ងៃលិច។ មិត្តរួមការងារម្នាក់ បានប្រគល់លិខិតមួយច្បាប់ ពីម្ចាស់ក្រុមហ៊ុនទៅឱ្យគាត់។ គាត់ត្រូវបានបណ្តេញចេញ។ លោករេនមានការភ្ញាក់ផ្អើលជាខ្លាំង។ គាត់មិនដឹងថា តើនឹងមានអ្វីកើតឡើងចំពោះគាត់នោះទេ។ គាត់ឆ្ងល់នៅក្នុង ចិត្តថា តើគាត់អាចឈរលើជំហររឹងមាំសម្រាប់ព្រះអម្ចាស់ដែរឬទេ បើសិនជាគាត់ត្រូវបានផ្ញើទៅគុក ឬ ប្រឈមមុខនឹងការស្លាប់។ ក្នុងនាមជាជនបរទេស គាត់គិតនៅក្នុងចិត្តថា តើគាត់អាចនឹងប្រកាន់ជំហរ រឹងមាំសម្រាប់ព្រះអម្ចាស់ដែរឬទេ ប្រសិនបើគាត់ត្រូវជាប់ពន្ធនាគារ ឬប្រឈមមុខនឹងការប្រហារជីវិតនោះ។ លោករេនបានអធិស្ឋានថា «ឱព្រះអម្ចាស់អើយ តើនេះជាពេលវេលា ដែលទូលបង្គំត្រូវត្រឡប់ទៅប្រទេស ភីលីពីនមែនទេ? បើសិនជាទ្រង់ចង់ឱ្យខ្ញុំស្លាប់នៅទីនេះ នោះប្រសើរជាងឱ្យទូលបង្គំស្លាប់នៅប្រទេសទូលបង្គំ។ មនុស្សជាច្រើនបានដឹងថា ទូលបង្គំកំពុងតែតយុទ្ធ ដើម្បីជំនឿរបស់ទូលបង្គំ»។ នៅព្រឹកថ្ងៃ បន្ទាប់ គាត់បានទៅការិយាល័យ ហើយបានផ្ញើអ៊ីមែលទៅម្ចាស់ក្រុមហ៊ុនរបស់គាត់។ គាត់បានសរសេរ ថា «ខ្ញុំគោរពមេណាស់ ដោយសារតែមេជាមិត្ត និងជាបងប្រុសធំរបស់ខ្ញុំ។ ខ្ញុំដឹងថា យើងត្រូវការពិភាក្សា អំពីបញ្ហាដែលខ្ញុំបានបង្កឡើង នៅពេលដែលមេចុះមកពិនិត្យការិយាល័យ»។ នៅយប់នោះ ម្ចាស់ក្រុមហ៊ុន របស់គាត់ បានចូលមកធ្វើការ។ ម្ចាស់ក្រុមហ៊ុនមើលទៅ ហាក់ដូចជាមានការខកចិត្តនោះទេ។ ប៉ុន្តែ ទន្ទឹមគ្នានេះ លោករេនក៏បានអធិស្ឋានថា «សូមទ្រង់ជួយទូលបង្គំផង។ សូមកុំឱ្យម្ចាស់ក្រុមហ៊ុនរបស់ទូលបង្គំ រករឿងទូលបង្គំឡើយ។ ប៉ុន្តែ បើសិនជាទ្រង់មានព្រះហឫទ័យចង់ឱ្យទូលបង្គំស្លាប់ នោះខ្ញុំនឹង សុខចិត្តស្លាប់»។

បន្ទាប់មក ម្ចាស់ក្រុមហ៊ុន បានហៅលោករេនមកជួបនឹងគាត់។ លោករេនបានព្យាយាមក្រោកឈរ ឡើង ប៉ុន្តែ មិនអាចធ្វើបាន។ ជង្គង់របស់គាត់ កក្រើកមិនកើត។ គាត់មានអារម្មណ៍ថា គាត់ហាក់ដូចជា មនុស្សខ្វិនអ៊ីចឹង។ គាត់បានលើកខ្លួនគាត់ចេញពីកៅអី ដោយប្រើកម្លាំងទាំងអស់របស់គាត់ ហើយដើរឱនៗ ទៅការិយាល័យនៃម្ចាស់ក្រុមហ៊ុនរបស់គាត់ទាំងបង្ខំ។ នៅពេលដែលគាត់លើកដៃរបស់គាត់គោះទ្វារ នោះគាត់បានឮសំឡេងថា «កុំភ័យអី ខ្ញុំគឺជាព្រះពិត។ ខ្ញុំស្មោះត្រង់នឹងការសន្យារបស់ខ្ញុំ»។

មួយភ្លែតក្រោយមក គាត់បានបើកទ្វារ ហើយបានឃើញមុខម្ចាស់របស់គាត់។ នៅពេលដែល គាត់ដើរចូលទៅ នោះគាត់បានឮសំឡេងខ្សាវៗចំនួន២ដងថា «ខ្ញុំគឺជាព្រះជាម្ចាស់ពិត។ ខ្ញុំស្មោះត្រង់ នឹងការសន្យារបស់ខ្ញុំ»។ សំឡេងនោះ បាននិយាយកាន់តែខ្លាំងឡើង។ ម្ចាស់ក្រុមហ៊ុនបានប្រគល់ក្រដាស ២សន្លឹកទៅឱ្យលោករេន រួចនិយាយថា «ឯងត្រូវរើសយកក្រដាស១»។

នៅពេលដែលលោករេន បង្ហាញទឹកមុខភ័យស្លន់ស្លោ នោះម្ចាស់ក្រុមហ៊ុនបាននិយាយថា «ក្រដាសសមួយ គឺជាកិច្ចសន្យាការងារ។ ក្រដាសមួយទៀត គឺជាសំបុត្រយន្តហោះ»។ លោករេនបាន ធូរចិត្ត និងបានអធិស្ឋាននៅក្នុងចិត្តថា «សូមអរគុណព្រះអម្ចាស់ ដែលក្រដាសទាំងនេះ មិនមែនក្រដាស ពន្ធនាគារ» គាត់បានរើសយកសំបុត្រយន្តហោះ។ ប៉ុន្តែ ម្ចាស់ក្រុមហ៊ុនបាននិយាយថា «សូមអានកិច្ចសន្យា នេះសិន»។ លោករេនបានមើលឃើញថា ប្រាក់ខែរបស់គាត់ នឹងត្រូវគុណនឹងពីរ បើសិនជាគាត់បានធ្វើការមួយម៉ោងនៅថ្ងៃសៅរ៍។ គាត់បានជ្រើសរើសសំបុត្រម្តងទៀត។ នៅពេលគាត់ ឆ្ងល់នៅក្នុងចិត្តថា តើគាត់នឹងបង់ប្រាក់វាយ៉ាងដូចម្តេច គាត់បានឮសំឡេងម្តងទៀត ដែលនិយាយថា «ខ្ញុំគឺជាព្រះអម្ចាស់។ ខ្ញុំស្មោះត្រង់នឹងការសន្យារបស់ខ្ញុំ»។ លោករេនបានត្រឡប់ទៅប្រទេសហ្វីលីពីនវិញ។ នៅលើយន្តហោះ គាត់បានអធិស្ឋានថា «ព្រះអង្គបានមានបន្ទូលថា «ខ្ញុំគឺជាព្រះជាម្ចាស់ពិត។ ខ្ញុំស្មោះត្រង់នឹងការសន្យា របស់ខ្ញុំ តើអ្វីទៅ ជាការសន្យារបស់អ្នកចំពោះខ្ញុំនោះ?»។ សំឡេងខ្សាវៗបាននិយាយថា «អ្នកមិនត្រូវ ចំណាយប្រាក់ទាំងអស់របស់ឯងនោះឡើយ រហូតទាល់តែឯងរកបានការងារថ្មី»។
