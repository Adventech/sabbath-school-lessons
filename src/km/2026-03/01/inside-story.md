### ការវិលត្រឡប់របស់អ្នកត្រួសត្រាយផ្លូវ

ដោយ Rick Kajiura

គ្មានអ្វីដែលគួរឱ្យរំភើបចិត្តជាងការទៅសួរសុខទុក្ខ ពួកអ្នកត្រួសត្រាយផ្លូវ នៃបេសកកម្មសកលនោះឡើយ។ អ្វីដែលកាន់តែរំភើបជាងនេះទៅទៀតនោះ គឺការបានឃើញពួកគេជួបជុំគ្នា ជាមួយក្រុមតូចៗនៃអ្នកជឿ ដែលទើបតែចាប់ផ្តើមដំបូង។ ខ្ញុំចាំបានថា ខ្ញុំធ្លាប់ធ្វើបែបនេះនៅក្នុងផ្ទះជាព្រះវិហារមួយ នៅតាមផ្លូវមានដីហុយទ្រលោម នៅក្នុងភូមិលើភ្នំមួយក្នុងប្រទេសឥណ្ឌូនេស៊ី និងធ្លាប់បានលួចចូលទៅក្នុងបន្ទប់ផ្ទះល្វែងតូចមួយ ក្នុងអគារថ្មបេតុងក្នុងប្រទេសមួយ នៅអឺរ៉ុបខាងកើត។ ខ្ញុំនៅចាំបាននូវសំឡេងតង់ពណ៌ខៀវ១ ឮសូរស្រិកស្រាក់ៗ នៅក្រោមជើង និងនៅលើក្បាលរបស់យើង ក្នុងតំបន់អាស៊ីអាគ្នេយ៍ និងបានឃើញអ្នកត្រួសត្រាយម្នាក់ កំពុងបង្រៀនមេរៀនថ្នាក់ថ្ងៃបរិសុទ្ធ នៅក្រោមដើមឈើក្នុងតំបន់ជនបទនៃទ្វីបអាហ្វ្រិក។

ប៉ុន្តែ ក្រុមជំនុំដែលខ្ញុំបានទៅសួរសុខទុក្ខកាលពីពេលថ្មីៗនេះ គឺខុសប្លែកពីគេទាំងស្រុង។ អ្នកត្រួសត្រាយទាំងនោះ បានចាកចេញទៅជាច្រើនឆ្នាំកន្លងមកហើយ។ លើកនេះ ខ្ញុំបានឃើញអគារព្រះវិហារមួយ ដែលមានសាលារៀននៅជាប់នោះ។ ក្រុមជំនុំនេះមានសមាជិកចំនួន ១២៥នាក់ ហើយសាលារៀននេះ ត្រូវបានចាត់ទុកថាជាសាលាដ៏ល្អបំផុតមួយនៅក្នុងទីក្រុង។

នៅពេលដែលអ្នកត្រួសត្រាយបានមកដល់ដំបូង មិនទាន់មានអ្នកអាត់វេនទីស្ទណាម្នាក់នៅក្នុងទីក្រុងនេះឡើយ។ ពួកគេបានចាប់ផ្តើមព្យាយាមលក់សៀវភៅ ដោយដើរគោះទ្វារពីផ្ទះមួយទៅផ្ទះមួយ។ ពួកគេបានព្យាយាមជួលកន្លែងសម្រាប់ការរៀបចំការជួបជុំ ប៉ុន្តែប្រជាជនមិនព្រមជួលឱ្យឡើយ។ ការនេះ ពួកគេយល់យ៉ាងច្បាស់ គឺថាពួកគេមិនត្រូវបានគេស្វាគមន៍នៅក្នុងទីក្រុងនេះទេ។ ផ្ទុយទៅវិញ ពួកគេបានទទួលការអញ្ជើញ ឱ្យទៅកាន់ទីប្រជុំជនមួយ ដែលនៅក្បែរនោះ។ នៅទីនោះ ពួកគេអាចជួលសាលមួយ និងចាប់ផ្តើមធ្វើការងាររបស់ពួកគេ។ មិនយូរប៉ុន្មាន ពួកគេបានបង្កើតក្រុមតូចមួយ ដែលមានសមាជិកទទួលបុណ្យជ្រមុជទឹកចំនួន ៥នាក់។ បន្ទាប់មក ពួកគេត្រូវបានអញ្ជើញឱ្យទៅទីប្រជុំជនមួយទៀត ហើយមិនយូរប៉ុន្មាន ពួកគេក៏មានក្រុមមួយដែលមានសមាជិក ៣៧នាក់។

វាបានចំណាយពេលជាច្រើនឆ្នាំ ទម្រាំតែពួកគេត្រូវបានគេទទួលស្គាល់នៅក្នុងទីក្រុងធំ និងអាចរកកន្លែងជួបជុំបាន។ ពួកគេបានចាប់ផ្តើមក្រុមតូចមួយ ហើយភ្លាមៗនោះពួកគេមានសមាជិក ៣០នាក់។ ចាប់តាំងពីពេលនោះមក ក្រុមជំនុំនេះបានបង្កើតហៅថា “ក្រុមតូច” ចំនួនបួនទៀត។ ក្រុមតូចៗនៅក្នុងទីប្រជុំជនក្បែរៗនោះ ក៏បានរីកចម្រើនទៅជាក្រុមជំនុំធំ ហើយក្រុមជំនុំនីមួយៗក៏បានចាប់ផ្តើមបង្កើតក្រុមជំនុំតូចៗជាបន្តបន្ទាប់ទៀតដែរ។ អ្នកត្រួសត្រាយដំបូងម្នាក់ ដែលបានជួយចាប់ផ្តើមក្រុមជំនុំទាំងនេះ បានត្រឡប់មកសួរសុខទុក្ខតំបន់នេះជាមួយខ្ញុំ។ ឥឡូវនេះ គាត់គឺជាប្រធានបេសកកម្មអាត់វេនទីស្ទប្រចាំតំបន់ ហើយគាត់បានបង្ហាញខ្ញុំ ពីរបៀបដែលព្រះបានប្រទានពរដល់ការខិតខំប្រឹងប្រែងរបស់គាត់ ក្នុងរយៈពេលជាច្រើនឆ្នាំកន្លងមកនេះ។

សព្វថ្ងៃនេះ អ្នកត្រួសត្រាយនៅតែបន្តទៅកាន់តំបន់ ដែលមិនទាន់មានអ្នកទៅដល់។ អ្នកត្រួសត្រាយដូចជាលោក តាហ្គូហ៊ិ (Taguhi) និងលោក អកវ៉ាន់ (Aghvan) កំពុងធ្វើការនៅក្នុងតំបន់ដែលមានអ្នកអាត់វេនទីស្ទតិចតួចបំផុត។ សូមអធិស្ឋានសម្រាប់អ្នកផ្សាយដំណឹងល្អ និងអ្នកត្រួសត្រាយបេសកកម្មសកល នៅពេលដែលពួកគេចាប់ផ្តើមការងារថ្មីនៅក្នុងតំបន់ ១០/៤០ តាមទីក្រុងធំៗក្នុងពិភពលោក និងក្នុងចំណោមមនុស្សដែលធ្លាប់ជាគ្រីស្ទានពីមុន។ សូមព្រះអង្គប្រទានពរដល់ការងាររបស់ពួកគេ នៅពេលដែលពួកគេចែកចាយដំណឹងល្អ និងចាប់ផ្តើមបង្កើតក្រុមថ្វាយបង្គំថ្មីៗ៕
