### ការជួបគ្នាដែលដឹកនាំដោយព្រះវិញ្ញាណ

(ឈ្មោះរបស់អ្នកនិពន្ធ តួអង្គ និងទីកន្លែងត្រូវបានរក្សាជាការសម្ងាត់)

ខ្ញុំ និងភរិយារបស់ខ្ញុំបានអធិស្ឋានយ៉ាងមុតមាំ សូមឱ្យព្រះជួយយើងឱ្យបង្កើតក្រុមជំនុំតាមផ្ទះមួយកន្លែង នៅក្នុងប្រទេសដែលរឹតបន្តឹងសេរីភាពសាសនា ជាកន្លែងដែលខ្ញុំកំពុងបម្រើការជាអ្នកផ្សាយដំណឹងល្អ។ យើងដឹងថា ដំបូងឡើយមានតែជនបរទេសប៉ុណ្ណោះ ដែលនឹងមកចូលរួម ប៉ុន្តែយើងមានជំនឿថា ព្រះនឹងបញ្ជូនអ្នកស្រុកក្នុងតំបន់នោះ ដែលទ្រង់បានរៀបចំចិត្តគំនិតរួចជាស្រេច ឱ្យមកឯយើង។ វាបានចំណាយពេលមួយរយៈ ប៉ុន្តែទ្រង់ពិតជាបានប្រទានឱ្យមែន!

ថ្ងៃមួយ ខ្ញុំបានជួបលោក សាល់ម៉ាន (Salman) នៅពេលដែលខ្ញុំបានហៅតាក់ស៊ីរបស់គាត់។ ខណៈពេលដែលគាត់កំពុងជូនខ្ញុំទៅកាន់គោលដៅ យើងចាប់ផ្តើមជជែកគ្នា ហើយមិនយូរប៉ុន្មាន យើងក៏បានបង្វែរទៅរកប្រធានបទដែលនិយាយអំពីព្រះ។ គាត់បានចែករំលែកថា គាត់បានក្លាយជាគ្រីស្ទាន កាលពីពីរឆ្នាំមុន បន្ទាប់ពីបានស្វែងរកចម្លើយចំពោះសំណួរខាងវិញ្ញាណរបស់គាត់តាមអ៊ីនធឺណិត។

ខ្ញុំ និងសាល់ម៉ាន បានបង្កើតមិត្តភាពជាមួយគ្នាបន្តិចម្តងៗ។ មានពេលមួយ ខ្ញុំបានសួរគាត់ថាតើគាត់មានក្រុមជំនុំ ឬមានមិត្តភក្តិជាគ្រីស្ទានដែរឬទេ? ខ្ញុំពិតជាមានការភ្ញាក់ផ្អើលយ៉ាងខ្លាំង នៅពេលដែលគាត់ប្រាប់ថា គាត់មិនដែលបានជួបអ្នកគ្រីស្ទានណាម្នាក់ពីមុនមកឡើយ។ ខ្ញុំ និងភរិយាមានអារម្មណ៍ថា ត្រូវតែអញ្ជើញគាត់មកចូលរួមការថ្វាយបង្គំនៅផ្ទះរបស់យើង ហើយគាត់ក៏បានចាប់ផ្តើមមកចូលរួមជាទៀងទាត់។ យើងមានសេចក្តីត្រេកអរណាស់ នៅពេលគាត់រៀបការហើយ គាត់បាននាំភរិយារបស់គាត់មកជាមួយ ហើយនាងក៏បានស្រឡាញ់ព្រះយេស៊ូវផងដែរ។

ថ្ងៃមួយ សាល់ម៉ាន បានសម្រេចចិត្តប្រាប់លោក ហ្វៃសាល់ (Faisal) ដែលជាបងប្រុសរបស់គាត់ អំពីសេចក្តីស្រឡាញ់របស់គាត់ចំពោះព្រះគ្រីស្ទ។ អ្វីដែលគួរឱ្យភ្ញាក់ផ្អើលនោះគឺ ហ្វៃសាល់ បានប្រាប់គាត់ថា គាត់ក៏ជាគ្រីស្ទានរួចទៅហើយដែរ។ គាត់បានពន្យល់ថា គាត់ក៏បានស្វែងរកចម្លើយចំពោះសំណួរខាងវិញ្ញាណតាមអ៊ីនធឺណិត ហើយបានប្រគល់ចិត្តជឿលើព្រះយេស៊ូវដូចគ្នា។

សាល់ម៉ាន បានប្រាប់ហ្វៃសាល់ អំពីក្រុមជំនុំតាមផ្ទះរបស់យើង ហើយហ្វៃសាល់ក៏បានមកចូលរួមជាមួយក្រុមយើង។ គាត់គ្មានការសង្ស័យឡើយថា គឺព្រះដែលបាននាំយើងឱ្យមកជួបគ្នា។ កាលពីពេលថ្មីៗនេះ គាត់បានទៅរកក្រុមជំនុំមួយ ដើម្បីចូលរួមថ្វាយបង្គំ ប៉ុន្តែអ្នកដឹកនាំនៅទីនោះបានប្រាប់គាត់ថា កុំត្រឡប់មកវិញអី ព្រោះច្បាប់ហាមឃាត់មិនឱ្យមនុស្ស ដែលមានជំនឿដូចហ្វៃសាល់ឯងនៅទីនោះឡើយ។

យើងបានសរសើរតម្កើងព្រះដែលបានជួយយើង ឱ្យមានកន្លែងមួយសម្រាប់ហ្វៃសាល់ថ្វាយបង្គំ។ បន្ទាប់ពីបានរក្សាភាពស្ងប់ស្ងាត់ និងអធិស្ឋានអស់រយៈពេលពីរឆ្នាំ សាល់ម៉ាន និងហ្វៃសាល់ បានសម្រេចចិត្តប្រាប់ក្រុមគ្រួសារអំពីជំនឿថ្មីរបស់ពួកគេ។ ដោយសារព្រះគុណនៃព្រះ ឪពុកម្តាយរបស់ពួកគេ បានទទួលយកការសម្រេចចិត្តនេះ ប៉ុន្តែសមាជិកគ្រួសារផ្សេងទៀតបានបដិសេធ។

សូមលោកអ្នកមេត្តាជួយអធិស្ឋានឱ្យ ហ្វៃសាល់ សាល់ម៉ាន និងភរិយារបស់គាត់ ឱ្យនៅតែបន្តមានទំនាក់ទំនងជាមួយព្រះយេស៊ូវ ហើយសូមឱ្យខ្ញុំ និងភរិយារបស់ខ្ញុំ បានជួបនូវការជួបជុំដែលដឹកនាំដោយព្រះវិញ្ញាណបែបនេះ ឱ្យបានកាន់តែច្រើនថែមទៀត ជាមួយកូនៗរបស់ព្រះនៅក្នុងប្រទេសដែលរិតត្បិតនេះ៕
