### បេសកកម្មចាប់ផ្តើមតាំងពីវ័យក្មេង

(by Kathie Lichtenwalter)

“ទោះបីជាកាលពីក្មេង ខ្ញុំបានអានរឿងទាំងអស់របស់លោក អេរីក ប៊ី. ហែរ (Eric B. Hare) ក៏ដោយ ក៏ខ្ញុំមិនដែលនឹកស្មានថា ខ្លួនឯងអាចក្លាយជាអ្នកផ្សាយដំណឹងល្អម្នាក់នោះដែរ” នេះគឺជាប្រសាសន៍របស់លោក ម៉ៃរ៉ុន អ៊ីសេមីងហ្គើ (Myron Iseminger) ដែលជាបេសកជនម្នាក់ កំពុងបម្រើការនៅប្រទេសលីបង់។

នៅចុងទសវត្សរ៍ឆ្នាំ១៩៧០ បងស្រីរបស់លោក ម៉ៃរ៉ុន បានប្រកាសថា នាងនឹងសម្រាកមួយឆ្នាំ ដើម្បីធ្វើជាគ្រូបង្រៀនភាសាអង់គ្លេសស្ម័គ្រចិត្ត នៅប្រទេសជប៉ុន។ បន្ទាប់ពីសិក្សានៅមហាវិទ្យាល័យបានមួយឆ្នាំក្នុងជំនាញទេវវិទ្យា លោក ម៉ៃរ៉ុន ក៏បានដាក់ពាក្យស្នើសុំធ្វើជាអ្នកស្ម័គ្រចិត្តដើម្បីយកបទពិសោធន៍ ទៅកាន់មជ្ឈិមបូព៌ា។ វាគឺជាការសម្រេចចិត្តដែលបានតម្រង់ទិសដៅជីវិតរបស់លោក ដែលនៅសេសសល់ទាំងប៉ុន្មាន។

លោក ម៉ៃរ៉ុន និយាយថា៖ “ខ្ញុំគិតថាវាពិតជាអស្ចារ្យណាស់ ក្នុងការទទួលបានបទពិសោធន៍នៅក្នុងទឹកដីនៃព្រះគម្ពីរ។ ក្រៅពីនេះ ខ្ញុំគិតថាវានឹងផ្តល់ឱ្យខ្ញុំនូវបទពិសោធន៍ក្នុងកិច្ចការបម្រើព្រះដ៏មានប្រយោជន៍។ ប៉ុន្តែខ្ញុំទទួលបានច្រើនជាងនេះទៅទៀត។ ក្នុងឆ្នាំនោះ ទស្សនៈរបស់ខ្ញុំបានចាប់ផ្តើមផ្លាស់ប្តូរ។ ខ្ញុំបានឃើញវប្បធម៌ផ្សេង របៀបគិតផ្សេង។ ខ្ញុំបានទៅលេងផ្ទះសិស្សរបស់ខ្ញុំ ហើយបានស្តាប់រឿងរ៉ាវ និងឃើញការឈឺចាប់របស់ពួកគេ។ ខ្ញុំចាប់ផ្តើមយល់ពីជីវិតតាមរយៈទស្សនៈរបស់មនុស្សម្នាក់ទៀត។ ខ្ញុំបានដឹងថា វាជារឿងងាយស្រួលណាស់ ក្នុងការវិនិច្ឆ័យមនុស្សមួយក្រុម ប៉ុន្តែនៅពេលដែលលោកអ្នកបានជួបពួកគេផ្ទាល់ម្នាក់ម្ដងៗ ទើបលោកអ្នកឈប់ថ្កោលទោសគេ ហើយបែរជាមានសេចក្តីស្រឡាញ់ពួកគេវិញ”។ លោក ម៉ៃរ៉ុន ចាប់ផ្តើមយល់ដឹងពីការត្រាស់ហៅ ឱ្យមកបម្រើព្រះជាម្ចាស់ក្នុងរយៈពេលវែងនៅក្រៅប្រទេស។

នៅពេលនោះ តម្រូវការសម្រាប់គ្រូគង្វាលដែលធ្វើការក្រៅប្រទេស ហាក់ដូចជាមិនសូវមានច្រើនទេ ប៉ុន្តែលោក ម៉ៃរ៉ុន បានសន្និដ្ឋានថា ប្រសិនបើក្រុមជំនុំទូទាំងពិភពលោក បានត្រៀមថវិកាសម្រាប់បេសកកម្ម នោះគេនឹងត្រូវការមន្ត្រីហិរញ្ញវត្ថុដើម្បីគ្រប់គ្រងថវិកាទាំងនោះ។ នៅពេលលោកត្រឡប់ទៅមហាវិទ្យាល័យវិញ លោកបានបន្ថែមជំនាញទីពីរផ្នែកធុរកិច្ច។ វាគឺជាចំណុចរបត់ដ៏សំខាន់មួយទៀត ដែលកំណត់ទិសដៅនៃជីវិតរបស់លោក។

ដំបូន្មានដែលអ្នកដឹកនាំក្រុមជំនុំម្នាក់ បានចែករំលែកជាមួយលោក ម៉ៃរ៉ុន ក៏ស្ថិតនៅក្នុងចិត្តរបស់លោកដែរថា៖ ប្រសិនបើអ្នកចង់បម្រើនៅក្នុងប្រទេសក្រៅ ចូរប្រាកដថាអ្នកត្រូវតែរៀបការជាមួយអ្នកដែលមានចក្ខុវិស័យដូចគ្នា! ភរិយារបស់លោក ម៉ៃរ៉ុន គឺអ្នកស្រី កាន់ដេស មិនដែលបានចេញក្រៅសហរដ្ឋអាមេរិកឡើយ នៅពេលដែលពួកគេរៀបការជាមួយគ្នា។ លោក ម៉ៃរ៉ុន និយាយថា៖ “ប៉ុន្តែនាងមានចិត្តចង់ទៅ”។ ពួកគេទាំងពីរបានរង់ចាំឱកាសការងារនៅក្រៅប្រទេសជាមួយគ្នា។

ថ្ងៃមួយ មន្ត្រីម្នាក់មកពីមហាសន្និបាត (GC) បានប្រាប់លោក ម៉ៃរ៉ុន ថា មានឱកាសការងារជាអ្នកគ្រប់គ្រងហិរញ្ញវត្ថុនៅមជ្ឈិមបូព៌ា។ មិនយូរប៉ុន្មាន លោក ម៉ៃរ៉ុន និងអ្នកស្រី កាន់ដេស បានទៅរស់នៅក្នុងផ្ទះល្វែងមួយ នៅការិយាល័យក្នុងប្រទេសអេស៊ីប។ វាគឺជាការងារដ៏លំបាក ប៉ុន្តែទទួលបានផលតបស្នងដ៏ល្អ។ លោក ម៉ៃរ៉ុន និយាយថា៖ “ខ្ញុំរីករាយដែលបានស្ថិតនៅជួរមុខនៃបេសកកម្ម។ ខ្ញុំចូលចិត្តការរួមចំណែក នៅកន្លែងដែលធនធានមានកម្រិត។ ខ្ញុំចូលចិត្តឃើញការផ្លាស់ប្តូរដែលខ្ញុំអាចធ្វើបាន ទោះបីជាវាមានការប្រឈមខ្លាំងក៏ដោយ”។

ក្រោយមក លោក ម៉ៃរ៉ុន បានបម្រើការនៅក្នុងការិយាល័យក្រុមជំនុំជាច្រើន។ អ្វីៗទាំងអស់នេះ ហាក់ដូចជាមិនអាចទៅរួចឡើយសម្រាប់ក្មេងប្រុសម្នាក់ ដែលធ្លាប់ជក់ចិត្តនឹងរឿងរ៉ាវរបស់លោក អេរីក ប៊ី. ហែរ ប៉ុន្តែលោកអាចធ្វើជាបន្ទាល់ថា៖ “ប្រសិនបើយើងបើកចិត្តឱ្យទូលាយសម្រាប់កន្លែងណាដែលព្រះជាម្ចាស់ត្រូវការយើង នោះទ្រង់នឹងរៀបចំទិសដៅជីវិតរបស់យើង បានល្អជាងអ្វីដែលយើងអាចរៀបចំផែនការដោយខ្លួនឯងទៅទៀត”៕
