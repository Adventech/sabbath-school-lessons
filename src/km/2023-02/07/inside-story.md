### ត្រេកអរឡើងវិញដោយសារព្រះគុណរបស់ព្រះ

កាលពីឆ្នាំ1970 ឪពុកម្តាយជនជាតិណាវ៉ាចូ (Navajo) 2គ្រួសារដែលមិនធ្លាប់ស្គាល់គ្នាពីមុនមកនោះ បានបញ្ជូនកូនៗរបស់ពួកគេឱ្យទៅស្នាក់នៅក្នុងអគារស្នាក់នៅមួយ នៅឯសាលាឥណ្ឌា អាត់វេនទីស្ទហូលប្រ៊ូខ (Holbrook) ដែលមានចម្ងាយ160គីឡូម៉ែតពីផ្ទះរបស់ពួកគេ។ នៅក្នុងសាលា ហូលប្រ៊ូខ ក្មេងប្រុសស្រីបានជួបគ្នា បន្ទាប់មកក្មេងៗទាំងនោះបានទទួលជឿព្រះយេស៊ូវ ព្រមទាំងទទួល បុណ្យជ្រមុជទឹកផងដែរ។ បន្ទាប់មក មានពិធីមង្គលការជាលើកដំបូងមួយ ត្រូវបានរៀបចំឡើងនៅក្នុង ព្រះវិហារសេវេនដេយអាត់វេនៃទីស្ទដែលទើបនឹងសាងសង់ហើយនៅក្នុងទីក្រុងឈីនលេ (Chinle) នៃ រដ្ឋអារីស្សូណា (Arizona)។

នៅពេលដែលកូនដំបូងរបស់លោកដេននីស (Dennis) និងនាងគ្លូរៀ ហ្វាល់ថុន (Gloria Fulton) បានកើតហើយនោះ ពួកគេក៏បានយកគាត់ទៅឯព្រះវិហារជាមួយនឹងពួកគេផងដែរ។ ដូច្នេះហើយ ទារកអូលីវើរ (Oliver) បានធំធាត់ឡើង ដោយដឹងថាព្រះវិហារអាត់វេនទីស្ទគឺជាព្រះវិហាររបស់គាត់។ ប៉ុន្តែ ទោះជាយ៉ាងនេះក្តី គាត់មិនមានបទពិសោធន៍ផ្ទាល់ខ្លួនជាមួយនឹងព្រះយេស៊ូវនោះឡើយ។ បញ្ហា ផ្សេងៗបានកើតមានឡើង ដូចជាបញ្ហាគិលានុបដ្ឋាយិការបស់នាងគ្លូរៀនៅឯមន្ទីរពេទ្យស្រុកជាដើម។ កាន់តែអាក្រក់ជាងនេះទៅទៀតនោះ គឺថា លោកដេននីសមានបញ្ហាជាមួយនឹងស្រា រីឯអូលីវើរនោះ ក៏ បានចាប់ផ្តើមផឹកស្រាតាំងពីក្មេងផងដែរ។

អូលីវើរបានបញ្ចប់ការសិក្សាពីសាលាវិទ្យាល័យ ហើយគាត់បានផ្លាស់ទៅឯទីក្រុងធំមួយ ដើម្បី បន្តសញ្ញាបត្រអនុបណ្ឌិតផ្នែកព័ត៌មានបច្ចេកទេសតទៅទៀត ប៉ុន្តែ គាត់បានដឹងថាជីវិតរបស់គាត់កំពុង តែត្រូវស្រាគ្រប់គ្រងហើយ។ នៅអាយុ38ឆ្នាំ អូលីវើរបានត្រឡប់ទៅឯស្រុកឈីនលីដែលជាកន្លែងដែល ម្តាយរបស់គាត់បានអធិស្ឋានឱ្យគាត់នោះ ដោយក្តីអស់សង្ឃឹម។ គាត់បានចាប់ផ្តើមចូលទៅឯព្រះវិហារ របស់គាត់កាលពីគាត់នៅក្មេងវិញ ដោយសង្ឃឹមថា គាត់នឹងបានផ្លាស់ប្តូរ។ អូលីវើរបានដឹងថាព្រះវិហារបាន ដឹកនាំបង្កើតការផ្សព្វផ្សាយដំណឹងល្អមួយទាក់ទងទៅនឹងការផ្តាច់គ្រឿងញៀន ដែលនៅក្នុងនោះ មាន មនុស្ស80នាក់ដែលមានបញ្ហាញៀនដូចជាគាត់ដែរ ពួកគេបានមកជួបជុំគ្នានៅក្នុងសាលប្រកបគ្នា5ថ្ងៃក្នុងមួយអាទិត្យ សម្រាប់កម្មវិធី «ព្រះយេស៊ូវ និងខ្ញុំ» ដែលបានយកគំរូទៅតាមកម្មវិធីផ្តាច់គ្រឿងញៀនផ្សេងៗ។ ជីវិតរបស់អូលីវើរបានចាប់ផ្តើមផ្លាស់ប្តូរ។ ទន្ទឹមគ្នានេះ ប៉ាស្ទ័រនៃព្រះវិហារនោះបានឃើញអូលីវើរ នៅក្នុងពួកជំនុំនៅថ្ងៃសប្បាតហ៍មួយ ហើយក៏បានអញ្ជើញគាត់ឱ្យញុាំអាហារថ្ងៃត្រង់នៅអាទិត្យបន្ទាប់ នោះផងដែរ។ អាហារថ្ងៃត្រង់បានផ្តល់ក្តីសង្ឃឹមដល់អូលីវើរ។ អូលីវើរបាននិយាយថា «នៅពេលដែល ខ្ញុំត្រឡប់មកផ្ទះវិញនោះ ខ្ញុំបានដឹងថា ខ្ញុំគឺជាបាបជនដ៏អាក្រក់ម្នាក់ដែលមិនអាចទៅស្ថានសួគ៌បានឡើយ ខ្ញុំគ្រាន់តែយល់ឃើញថា ប្រសិនបើខ្ញុំមានអារម្មណ៍ស្ងប់នោះ ខ្ញុំនឹងអាចជួយអ្នកដទៃខ្លះឱ្យឡើង ស្ថានសួគ៌បាន ប៉ុន្តែ នៅពេលថ្ងៃត្រង់នោះ ប៉ាស្ទ័របានប្រាប់ខ្ញុំវិញថា អំពើបាបរបស់ខ្ញុំអាចនឹងលើកលែង ទោសឱ្យបាន។ ព្រះយេស៊ូវនឹងទទួលយកអ្នកតាមរបៀបអ្នក។ ខ្ញុំបានភ្ញាក់ផ្អើលយ៉ាងខ្លាំង ពីព្រោះ ពាក្យទាំងនេះបានផ្តល់ក្តីសង្ឃឹមដល់ខ្ញុំ»។

ឥឡូវនេះ អូលីវើរបានផ្តាច់ស្រាបាន4ឆ្នាំហើយ។ នៅឯព្រះវិហារ អូលីវើរបានជួបនឹងស្ត្រីម្នាក់ ឈ្មោះថាត្រាស៊ី (Traci) នាងក៏មានរឿងផ្ទាល់ខ្លួនដែលព្រះយេស៊ូវបាន រំដោះនាងចេញពីការញៀនហេរ៉ូអ៊ីនផងដែរ។ ប៉ាស្ទ័របានជ្រមុជទឹកឱ្យ អូលីវើរមុនពេលដែលអូលីវើរ និងត្រាស៊ីបានរៀបការនៅក្នុងព្រះវិហារ តែមួយ ដែលឪពុកម្តាយរបស់គាត់បានរៀបការនឹងគ្នាកាលពី40ឆ្នាំ មុន។
