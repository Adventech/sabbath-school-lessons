### ទីបន្ទាល់ ដែលមិនទាន់មានប្រសិទ្ធភាព

លោកគីម សុន ដែលជាយុវជននៃប្រទេសកូរ៉េខាងត្បូង បានសិក្សានៅឯសាកលវិទ្យាល័យ អាត់វេនទីស្ទ នៅក្នុងប្រទេសហ្វីលីពីន ដោយគិតថា តើគាត់អាចរ័កបានការងារ ចេញពីការដើរពីផ្ទះមួយ ទៅផ្ទះមួយនេះបានដែរឬទេ បន្ទាប់ពីគាត់បានឃើញបុរសម្នាក់ ដែលគាត់បានអញ្ជើញ ឱ្យមកចូលរួមកម្មវិធី ផ្សាយដំណឹងល្អ បានទទួលបុណ្យជ្រមុជទឹកនៅក្នុងកម្មវិធីនោះ។ លោកសុនមិនមែនជាគ្រីស្ទាននោះទេ ហើយការដើរពីផ្ទះមួយទៅផ្ទះមួយទៀត ក៏ជាកិច្ចការដំបូងរបស់គាត់ដែរ។ លោកសុនបានសួរនាំប៉ាស្ទ័រ ម្នាក់ ដែលបានដើរសួរសុខទុក្ខពីផ្ទះមួយទៅផ្ទះមួយទៀត ជាមួយសិស្សដទៃទៀតថា «លោកគ្រូ តើយើង ដើរសួរសុខទុក្ខពីផ្ទះមួយទៅផ្ទះមួយទៀត ដើម្បីអ្វី? តើសកម្មភាពបែបនេះឯង ដែលគេបានហៅថា ជាការ បម្រើសហគមន៍នោះ?»។ ប៉ាស្ទ័របានញញឹម រួចនិយាយថា «សកម្មភាពបែបនេះ មិនមែនជាការបម្រើ សហគមន៍នោះទេ ប៉ុន្តែ ពួកយើងបានហៅសកម្មភាពបែបនេះថា ជាបេសកកម្មវិញ។ លោកសុនបាន និយាយទៀតថា «បាទលោកគ្រូ សកម្មភាពបែបនេះឯង ដែលគេហៅថា ការបម្រើការជាបេសកជន បើ ដូច្នេះមែន ខ្ញុំសូមធ្វើការងារនេះដែរ»។ ប៉ាស្ទ័របានតបមកវិញថា «សុនអាចធ្វើការងារនេះបាន គ្រាន់តែ ថា ប្រាក់ចំណូលមិនសូវជាបានច្រើននោះទេ»។ សុនបានសួរទៀតថា «តើខ្ញុំអាចធ្វើកិច្ចការនេះ ដោយ របៀបណា?»។ ប៉ាស្ទ័របានតបមកវិញថា «បើសិនជាសុនចង់ធ្វើជាបេសកជន់ សុនអាចផ្លាស់ប្ដូរមុខវិជ្ជា គិលានុប្បដ្ឋាក មកផ្នែកទេវវិទ្យាវិញ»។ លោកសុនបាននិយាយថា «បាទលោកគ្រូ ចាំខ្ញុំសួរនាំម៉ាក់ខ្ញុំសិន»។

លោកសុនបានមកសិក្សានៅក្នុងប្រទេសហ្វីលីពីន ដោយសារតែឳពុកម្តាយរបស់គាត់ មាន បំណងចង់ឱ្យគាត់ធ្វើអ្វីមួយ នៅក្នុងជីវិតរបស់គាត់។ កាលពីមុន គាត់ធ្លាប់តែបានរស់នៅ ដើម្បីខ្លួនគាត់។ ប៉ុន្តែ នៅពេលដែលលោកសុនសួរនាំម្តាយរបស់គាត់ ក្រែងលោគាត់អាចផ្លាស់ប្ដូរមុខវិជ្ជា ដែលគាត់កំពុងតែ សិក្សា។ ម្តាយរបស់គាត់ មិនយល់ពីអ្វីដែលគាត់បានស្នើសុំនោះទេ ម្ល៉ោះហើយ គាត់បានសួរថា «កូនប្រុស តើមុខវិជ្ជាទេវវិទ្យាគឺជាអ្វី?»។ លោកសុនបាននិយាយថា «មុខវិជ្ជាទេវវិទ្យា គឺជាការបម្រើព្រះវិហារ» ដោយ ប្រាប់បន្ថែមថា ការបង់ថ្លៃសិក្សា អាចនឹងមានតម្លៃទាប។ ស្ត្រីដែលជាម្តាយ បានប្រាប់លោកសុនថា គាត់ អាចយកមុខវិជ្ជានេះបាន ប៉ុន្តែ ទន្ទឹមគ្នានេះ គាត់ក៏ត្រូវតែរៀនមុខវិជ្ជាគិលានុប្បដ្ឋាករបស់គាត់ ឱ្យបាន ចប់សព្វគ្រប់ផងដែរ។ លោកសុនបានតបទៅវិញថា «ប៉ុន្តែ គិលានុប្បដ្ឋាក ដូចជាមិនសូវសំខាន់សម្រាប់ខ្ញុំសោះ»។ នៅពេលនោះ ស្ត្រីដែលជាម្តាយ បាននឹកឃើញគំនិត។ តាមពិតទៅ គោលដៅរបស់នាង មិនមែន ចង់ឱ្យគាត់មានបាននោះទែ ប៉ុន្តែ នាងគ្រាន់តែចង់ឱ្យគាត់ក្លាយទៅជាមនុស្សល្អប៉ុណ្ណោះ។ នាងបានប្រាប់គាត់ថា «បើសិនជាកូនបានរើសមុខវិជ្ជានេះ តើកូនអាចផឹកស្រា ឬជក់បារីបានដែរ ឬទេ?»។ នៅពេលដែលលោកសុនឆ្លើយថា ទេ នោះនាងបានឧទានថា «បើអ៊ីចឹងមែន កូនអាចផ្លាស់ប្ដូរបាន»។

លោកសុនស្រឡាញ់មុខវិជ្ជាទេវវិទ្យាណាស់។ គាត់បានរៀនអំពីមូលដ្ឋានគ្រឹះនៃព្រះគម្ពីរ ដែល បានចែងអំពីថ្ងៃសប្បាតហ៍ ដែលជាថ្ងៃទី៧។ គាត់បានអានបទទំនាយ នៅក្នុងគម្ពីរដានីយ៉ែល និងគម្ពីរ វិវរណៈ។ បន្ទាប់មក គាត់បានទទួលបុណ្យជ្រមុជទឹក។ នៅពេលដែលគាត់ត្រឡប់មកផ្ទះវិញ នៅពេល វិសម្សកាល នោះគាត់បានព្យាយាមបញ្ចុះបញ្ចូលឪពុកម្តាយរបស់គាត់ ឱ្យទទួលយកគោលជំនឿថ្មីរបស់ គាត់។ លោកសុនបាននិយាយថា «ប៉ា ម៉ាក់ សូមលោកទាំង២មកអង្គុយនៅទីនេះសិន ខ្ញុំមានរឿងចង់ពិភាក្សា។ តាមពិតទៅ ថ្ងៃអាទិត្យមិនមែនជាថ្ងៃសប្បាតហ៍នោះទេ ប៉ុន្តែ គឺថ្ងៃសៅរ៍វិញទេ។ តើប៉ាម៉ាក់បានដឹងអំពីការដែលលោកដានីញ៉ែល បានបកស្រាយសុបិនរបស់ស្តេចនេប៊ូក្នេសាដែរឬទេ?»។

លោកសុនមិនយល់ថា ហេតុអ្វីបានជាឪពុកម្តាយរបស់គាត់ មិនព្រមបើកចិត្តទទួលសេចក្ដី ពិតនៃថ្ងៃសប្បាតហ៍នេះឡើយ។ គាត់បានឈប់និយាយ នៅពេលដែលម្ដាយរបស់គាត់ បានសូមឱ្យ គាត់ឈប់និយាយ ដោយប្រាប់គាត់ថា «កូនជឿព្រះតាមបែបនេះចុះ ប៉ុន្តែ ប៉ាម៉ាក់ជឿតាមបែបផ្សេង»។ លោកសុនបានតបទៅវិញថា «ប៉ុន្តែ ព្រះដែលពួកយើងនេះ គឺជាព្រះតែមួយទេ ដូច្នេះ ហើយ យើងគួរតែធ្វើតាមព្រះបន្ទូលរបស់ព្រះ»។
