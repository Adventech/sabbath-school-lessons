### សោកនាដកម្មអំពីទឹកភ្នែក និងអាវរងា

នាងស៊ីតស៊ី សុនដូ (Tsitsi Zondo) សឹងតែមិនជឿអ្វី ដែលនាងបានឃើញ។ នៅពេលនោះ គឺជាយប់ដ៏ងងឹតមួយ នៅក្នុងប្រទេសស៊ីមបាវ៉េ (Zimbabwe) ប៉ុន្តែ នាងបានឃើញអាវរងា ដែលជាទំនិញ ដ៏មានតម្លៃរបស់នាង ជ្រុះរាយប៉ាយនៅតាមផ្លូវ ហើយមានមនុស្សកំពុងតែរើសយកអាវរងាទាំងនោះយ៉ាង ចលាចល។

នាងបានស្រក់ទឹកភ្នែក ដោយសារតែនាងបានខ្ចីប្រាក់ទិញអាវរងាទាំងនោះ នៅក្នុងប្រទេស អាហ្វ្រិកខាងត្បូង ប៉ុន្តែ ឥឡូវនេះ នាងមិនអាចបង់ប្រាក់កម្ចីបានឡើយ។ នាងមិនសល់អ្វីសោះ។ តើនាង ត្រូវផ្គត់ផ្គង់កូនរបស់នាងដោយរបៀបណា? នាងបានអធិស្ឋានថា «ឱព្រះអម្ចាស់ ទ្រង់បានដឹងហើយថា ទូលបង្គំកំពុងតែជួបនឹងការលំបាក នៅក្នុងការធ្វើឱ្យជំនួញរបស់ទូលបង្គំ មានភាពជោគជ័យ ប៉ុន្តែ ឥឡូវនេះ ហេតុការបែបនេះ បានកើតឡើងវិញ។ កាបូបអ្នកផ្សេងមិនជ្រុះ ហេតុអ្វីបានជាជ្រុះតែអីវ៉ាន់ទូលបង្គំ យ៉ាងដូច្នេះ?»។

នាងស៊ីតស៊ីបានខ្ចប់អាវរងាចំនួន៦០ ដាក់នៅក្នុងកាបូបដ៏ធំមួយ បន្ទាប់មក នាងបានចង កាបូបនោះ នៅលើដំបូលឡានក្រុង ដើម្បីធ្វើដំណើរពីព្រំប្រទល់នៃប្រទេសស៊ីមបាវ៉េ ជាមួយនឹងប្រជាជន ពីប្រទេសអាហ្វ្រិកខាងត្បូង ទៅកាន់ទីក្រុងប៊ូឡាវ៉ាយ៉ូ (Bulawayo) ដែលជាស្រុកកំណើតរបស់នាង។ ប៉ុន្តែ នៅពាក់កណ្តាលផ្លូវ មានអ្នកដំណើរម្នាក់ បានចេញពីឡានក្រុងនោះ ប៉ុន្តែ នៅពេលដែលរកមើលកាបូបរបស់នាង នោះអ្នកបើកបរបានបើករកមើលកាបូបរបស់នាង ដែលនៅលើដំបូលឡានក្រុង។ ប៉ុន្តែអ្នកបើកបរ មិនបានបិទកេះទំនិញរបស់នាងឡើយ។

ឡានក្រុងបានធ្វើដំណើរបន្តទៅទៀត ប៉ុន្តែ បន្ទាប់ពីបានធ្វើដំណើរមួយស្របក់រួចមក នោះ ឡានក្រុងបានឈប់នៅត្រង់ភ្លើងស្ដុប។ បន្ទាប់មក អ្នកបើកបរបានប្រកាសថា ឡានក្រុងបានជ្រុះអាវរងា តាំងពីពេលដែលអ្នកដំណើរម្នាក់នោះ បានចុះពីរថយន្ត។ នាងស៊ីតស៊ីបានស្លុតចិត្ត នៅពេលដែលអាវរងា នៅក្នុងកាបូបរបស់នាង នៅសល់ត្រឹមតែ១០។ អ្នកបើកបរឡានក្រុងបានដឹងថា គាត់ត្រូវទទួលខុសត្រូវប៉ុន្តែ គាត់បានលើកដៃ ទាំងដែលគាត់មិនដឹងថា គាត់ត្រូវធ្វើយ៉ាងណានោះទេ។ គាត់បានប្រាប់នាងស៊ីតស៊ី ថា «ឱ្យខ្ញុំសុំទោស ខ្ញុំមិនមានប្រាក់សងអ្នកទេ»។

បន្ទាប់ពីអធិស្ឋានរួច នាងស៊ីតស៊ីបានជូតទឹកភ្នែក ពីព្រោះនាងតែងតែបង្វិលតង្វាយ១០ហូត១ និងតង្វាយសទ្ធា ចេញពីទំនិញ ដែលនាងបាននាំមកពីប្រទេសអាហ្វ្រិកខាងត្បូង ដូច្នេះហើយ នាងប្រាកដនៅក្នុងចិត្តថា ព្រះនឹងយកព្រះទ័យទុកដាក់ចំពោះនាង។ នៅក្នុងទីក្រុងប៊ូឡាវ៉ាយ៉ូ មិត្តភក្តិបានឮពីស្ថានភាពរបស់នាង បន្ទាប់មក ពួកគេបានមកសួរ សុខទុក្ខនាង ដើម្បីអធិស្ឋានឱ្យនាង។ នៅពេលនោះ មិត្តភក្តិម្នាក់បានឱ្យនាងខ្ចីប្រាក់ ដើម្បីឱ្យនាងត្រឡប់ទៅកាន់អាហ្រិ្វកខាងត្បូងវិញ ដើម្បីទិញអាវរងារបន្ថែមទៀត។ នាងស៊ីតស៊ីបានទទួលយកប្រាក់ពីមិត្តភក្តិ របស់នាង ដោយចាត់ទុកថា ប្រាក់នោះ គឺជាចម្លើយឆ្លើយតបទៅនឹងការអធិស្ឋានរបស់នាង បន្ទាប់មក នាងបានត្រឡប់ទៅប្រទេសអាហ្វ្រិកខាងត្បូងវិញ។ នាងស៊ីតស៊ីអាចដឹកជញ្ជូនទៅតាមការកុម្មង់ទាំងអស់។ នាងត្រូវចំណាយការធ្វើដំណើរចំនួន៤លើក ដើម្បីឱ្យនាងបានសងប្រាក់ ដល់ម្ចាស់បំណុលទាំង២។ ប៉ុន្តែ នៅក្នុង់អំឡុងពេលនោះ នាងស៊ីតស៊ីតែងតែមានប្រាក់គ្រប់គ្រាន់ សម្រាប់អាហារ និងសម្រាប់ការចាយវាយ នៅក្នុងគ្រួសារ។ នាងតែងតែរក្សានូវការបង្វិលតង្វាយ១០ហូត១ និងការថ្វាយតង្វាយសទ្ធា។

នាងស៊ីតស៊ីមិនបានយល់ពីរបៀបដែលព្រះបានធ្វើការនោះទេ ប៉ុន្តែ នាងហ៊ាននិយាយថា ទ្រង់តែងតែផ្គត់ផ្គង់នាងជានិច្ច។ នាងបាននិយាយថា «បើទោះបីជាខ្ញុំ បានបាត់បង់អ្វីៗគ្រប់យ៉ាងក៏ដោយ ក៏ខ្ញុំ និងគ្រួសារខ្ញុំ មិនដែលស្រេកឃ្លានដែរ។ នៅលើតុរបស់យើង តែងតែមានអាហារ ជានិច្ច។ ដូច្នេះហើយ បើសិនជាលោកអ្នកស្មោះត្រង់ និងបានថ្វាយតង្វាយ១០ហូត១ និងតង្វាយសទ្ធា នោះទ្រង់តែងតែយាងមកជួយលោកអ្នកជានិច្ច»។
