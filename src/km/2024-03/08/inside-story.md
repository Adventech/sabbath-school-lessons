### គេហដ្ឋានដែលបានក្លាយទៅជាព្រះវិហារ

ឪពុកបានរក្សាពាក្យសម្តីរបស់គាត់ អំពីការរៀបចំកម្មវិធីថ្វាយបង្គំ នៅឯផ្ទះ នៅក្នុងប្រទេស អាមេនី។ ដោយបានហាមឃាត់ម្តាយ និងនាងអានុស មិនឱ្យទៅព្រះវិហារសេវេនដេយអាត់វេនទីស្ទ នៅព្រឹកថ្ងៃសប្បាតហ៍ឪពុកបានហៅពួកគេ ឱ្យទៅបន្ទប់ទទួលភ្ញៀវ។ សម្រាប់ថ្នាក់សប្បាតហ៍ស្គូល ពួកគេ បានសិក្សាមគ្គុទេសក៍នៃមេរៀនសប្បាតហ៍ស្គូល សម្រាប់មនុស្សចាស់ បន្ទាប់មក ពួកគេបាននាំគ្នាអធិស្ឋាន។ បន្ទាប់ ពីអធិស្ឋានរួច នាងអានុសបានចែកចាយសេចក្តីអធិប្បាយមួយយ៉ាងខ្លី។

កម្មវិធីថ្វាយបង្គំ បានបន្តអស់រយៈជាច្រើនខែ។ ឪពុក ដែលមិនធ្លាប់ទៅលេងព្រះវិហារអាត់-វេនទីស្ទ បានយកចិត្តទុកដាក់ចំពោះកម្មវិធីថ្វាយបង្គំ ដែលឱ្យតែមានភ្ញៀវចូលក្នុងផ្ទះរបស់គាត់ គាត់តែងតែ អញ្ជើញពួកគេ ចូលទៅក្នុងបន្ទប់ទទួលភ្ញៀវជានិច្ច បន្ទាប់មក គាត់បានបើកព្រះគម្ពីររបស់គាត់ រួចនិយាយ ថា «សូមស្វាគមន៍ មកកាន់កម្មវិធីថ្វាយបង្គំព្រះរបស់យើង»។ ថ្ងៃនេះ ជាថ្ងៃសប្បាតហ៍ ដូច្នេះហើយ អ្នក អាចចូលរួមថ្វាយបង្គំ ជាភ្ញៀវបាន។ ភ្ញៀវបានភ្ញាក់ផ្អើល និងងឿងឆ្ងល់ថា តើមានរឿងអ្វីកើតឡើង។ ឪពុក បានដឹងថា គាត់មិនចេះព្រះគម្ពីរនោះទេ។ ដូច្នេះហើយ គ្រួសារបានអានពីរបៀប ដែលព្រះយេស៊ូវបានជួបនឹងការល្បួងគ្រប់បែបយ៉ាងរបស់សាតាំង ដោយប្រើពាក្យថា «ដ្បិតមានសេចក្តីចែងទុកមកថា»។ ឪពុក បានលួចសរសើរ នៅក្នុងចិត្ត។ គាត់បានយល់ឃើញថា ប្រសិនបើគាត់មិនបានស្គាល់ព្រះគម្ពីរទេ នោះសាតាំងក៏អាចនឹងល្បួងគាត់ផងដែរ។ ចាប់តាំងពីថ្ងៃនោះមក គាត់បានចាប់ផ្តើមអានព្រះគម្ពីរជាប្រចាំ។ ខណៈ ពេលដែលគាត់អានព្រះគម្ពីរ នោះគាត់ក៏បានរកឃើញចម្លើយ ឆ្លើយតបទៅនឹងមូលហេតុ ដែលគាត់ និង ក្រុមគ្រួសារ កំពុងតែថ្វាយបង្គំនៅថ្ងៃសៅរ៍ ដែលជាថ្ងៃទី7 ខណៈពេលដែលគ្រីស្ទានជាច្រើននាក់ នៅក្នុង ប្រទេសអាមេនី បានថ្វាយបង្គំនៅថ្ងៃអាទិត្យ ដែលជាថ្ងៃទី1នោះ។ ឪពុកបានស្បថថា នាងអានុស និង ម្តាយ មិនអាចត្រឡប់ទៅព្រះវិហារអាត់វេនទីស្ទបានឡើយ ដូច្នេះហើយ បានជាគាត់ចង់រក្សាពាក្យសម្តី របស់គាត់។ នាងអានុសនឹកកម្មវិធី នៅឯព្រះវិហារណាស់ ប៉ុន្តែ នាងបានលាក់អារម្មណ៍របស់នាង ពីព្រោះ នាងបានយល់ថា កាតព្វកិច្ចរបស់នាង គឺដើម្បីស្រឡាញ់ឪពុករបស់នាង និងរង់ចាំឱ្យព្រះនាំគាត់ ឱ្យប្រែ ចិត្តវិញ។ ប៉ុន្តែ នៅពេលដែលនាងបានដឹងថា មានព្រះវិហារអាត់វេនទីស្ទនៅតាមផ្ទះ នៅក្នុងទីក្រុងរបស់ នាង កំពុងតែរៀបចំកម្មវិធីលាងព្រះអម្ចាស់ នៅថ្ងៃសប្បាតហ៍ នោះនាងបានសូមការអនុញ្ញាតពីឪពុក ដើម្បី ឱ្យនាងបានចូលរួម។ ប្រទេសអាមេនី គឺជាសង្គមគោរពដូនតា ដោយមានឪពុកជាច្រើននាក់ ជាអ្នកធ្វើការ សម្រេចចិត្ត នៅក្នុងផ្ទះ។ នាងបានសួរថា «តើប៉ាអាចអនុញ្ញាត ឱ្យខ្ញុំចូលរួមនៅក្នុងកម្មវិធីនេះ បានដែរ ឬទេ?»។ ឪពុកបានឆ្លើយថា «បុណ្យលាងព្រះអម្ចាស់? កូនស្រីដឹងតើថា ប៉ាក៏អាចដឹកនាំកម្មវិធីនេះ នៅក្នុង ផ្ទះរបស់យើងបានដែរ»។

បន្ទាប់មក ឪពុក និងម្តាយ បានក្លាយមកជាជីដូន និងជីតា។ ពីព្រោះនាងអានុសមានបងស្រី ម្នាក់ ដែលបានរៀបអាពាហ៍អាពិពាហ៍ និងបានចាកចេញពីផ្ទះ ហើយនាងក៏បានផ្តល់កំណើត ឱ្យក្មេងម្នាក់ ផងដែរ។ ម្តាយ (ភរិយារបស់ឪពុក) បានដឹងថា ព្រះវិហារបានថ្វាយកូននោះ និងក្រុមគ្រួសាររបស់គាត់ ដល់ព្រះ តាមរយៈការអធិស្ឋាន។ នាងបានប្រាប់ឪពុកថា «នៅក្នុងព្រះវិហារ ពួកគេបានអធិស្ឋានឱ្យពួកយើង ដូច្នេះហើយ បានជាខ្ញុំចង់យកកាដូ ទៅជូនពួកគាត់ ដើម្បីជាការតបស្នង»។

សេចក្តីសប្បុរសរបស់សមាជិកក្រុមជំនុំ បានប៉ះពាល់ចិត្តឪពុក បន្ទាប់មក គាត់បានអនុញ្ញាត ឱ្យនាងអានុស និងម្តាយត្រឡប់ទៅព្រះវិហារវិញ។
