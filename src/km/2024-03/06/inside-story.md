### ការទូលសូមសេចក្តីសង្គ្រោះឱ្យឪពុក

នៅក្នុងនាមជាសិស្សសាកលវិទ្យាល័យ នាងអានើសបានតាមដានខ្សែភាពយន្តមុិកសុិក អំពី ក្មេងប្រុសដ៏តូចម្នាក់ ដែលបានអធិស្ឋាន សូមឱ្យឪពុកបានកែប្រែចិត្ត។ នៅក្នុងខ្សែភាពយន្តនេះ ក្មេងប្រុស ម្នាក់នេះ បាននិយាយថា «ខ្ញុំជឿថា ប្រសិនបើខ្ញុំបានអធិស្ឋាន ឱ្យឪពុករបស់ខ្ញុំជារៀងរាល់ថ្ងៃ នោះគាត់ នឹងចូលមករកព្រះយេស៊ូវ តាមវិធីណាមួយ ជាមិនខាន។ ក្មេងប្រុសម្នាក់នេះ បានអធិស្ឋានជារៀងរាល់ថ្ងៃ ក្រោយមក ឪពុករបស់គាត់ បានថ្វាយចិត្តរបស់គាត់ដល់ព្រះ។ ដោយមានការបំផុសគំនិតពីសាច់រឿងនេះ នាងអានើសបានសម្រេចចិត្ត អធិស្ឋានជារៀងរាល់ថ្ងៃ ដើម្បីឱ្យឪពុករបស់នាង ជ្រមុជទឹក។ នាងនៅតែ បន្តអធិស្ឋានឱ្យគាត់ អស់រយៈពេល2ឆ្នាំ បន្ទាប់ពីនាងបានជ្រមុជទឹករួច។ នាងបានដឹងច្បាស់ នៅក្នុង ចិត្តថា គាត់នឹងចូលមករកព្រះ ជាមិនខាន។ ប៉ុន្តែ នៅពេលដែលការតានតឹងបានចាប់ផ្តើមកើតមាននៅក្នុងផ្ទះ នោះនាងបែរជាងឿងឆ្ងល់ថា តើនាងនឹងត្រូវរង់ចាំយូរប៉ុនណា។

បន្ទាប់ពីនាងបានជ្រមុជទឹករួច នាងអានើសបានក្លាយជាសមាជិកសកម្ម នៅក្នុងព្រះវិហារសេវេនដេយអាត់វេនទីស្ទ។ នាងបានធ្វើការស្ម័គ្រចិត្ត នៅក្នុងការផ្តួចផ្តើមកម្មវិធីព្រះវិហារ ដែលមានពេល ខ្លះ នាងទទួលបានប្រាក់ខែបន្តិចបន្តួច ហើយមានពេលខ្លះទៀត នាងមិនមានអ្វីទាល់តែសោះ។ ឪពុក ក៏មិនបានត្អូញត្អែរអ្វីដែរ ដោយសារតែគាត់បានអនុញ្ញាត ឱ្យនាងអានើសទៅព្រះវិហារ និងទទួលបុណ្យ ជ្រមុជទឹក។ ប្រទេសអេមេនី (Armenia) គឺជាសង្គមដែលមានមនុស្សភាគច្រើន គោរពដូនតា ដែលឪពុក ភាគច្រើន គឺជាអ្នកសម្រេចចិត្ត នៅក្នុងគ្រួសារ។ ប៉ុន្តែ ឪពុកចង់ឱ្យកូនស្រីរបស់គាត់ មានអនាគតល្អប្រសើរ ដូច្នេះហើយ គាត់មិនយល់ទាល់តែសោះថា ហេតុអ្វីបានជានាងធ្វើការ ដើម្បីប្រាក់បន្តិចបន្តួចយ៉ាងដូច្នេះ។ គាត់បាននិយាយថា «ព្រះវិហាររបស់កូន កំពុងតែប្រើប្រាស់កូនហើយ កូនជាមនុស្សមានសមត្ថភាពអ៊ីចឹង ពួកគេប្រាកដជាប្រើប្រាស់កូន ដោយគេងប្រវ័ញ្ចកម្លាំងពលកម្មកូនស្រីហើយ»។ នាងអានើសបាប់ផ្តើម មានអារម្មណ៍តានតឹង នៅពេលដែលនាងនៅក្នុងផ្ទះ។ នាងមិនចូលចិត្តបរិយាកាសបែបនេះឡើយ។ នាងបានសូមការអនុញ្ញាតពីឪពុក នៅពេលណាដែលគេអញ្ជើញនាង ឱ្យចូលរួមកម្មវិធីព្រះវិហារ។ ឪពុក បានអនុញ្ញាតឱ្យនាងទៅ ប៉ុន្តែ រាល់ពេលដែលនាងទៅ គាត់តែងតែរអ៊ូរទាំជានិច្ច។

នាងអានើសបានសម្រេចចិត្តថា នាងនឹងបើកចំហចិត្ត ទូលសូមទៅព្រះត្រង់ៗ ឱ្យអស់ពីចិត្ត»។ នាងទូលថា «ឱព្រះអម្ចាស់អើយ ទូលបង្គំដឹងថា ឪពុករបស់ទូលបង្គំ នឹងចូលមករកទ្រង់ ប៉ុន្តែ ទូលបង្គំ មានអារម្មណ៍ថា អស់កម្លាំងណាស់ ដូច្នេះហើយ ទូលបង្គំសូមឱ្យជម្រើស2យ៉ាងទៅទ្រង់ គឺថា បើសិនជា គាត់មិនបានចូលទៅរកទ្រង់ទេ នោះសូមឱ្យទ្រង់ចូលមករកគាត់វិញ។

ក្រោយមក នាងបានប្រាប់ម្តាយនាងថា «ថ្ងៃនេះ កូនបានអធិស្ឋាន ទូលសូមទៅព្រះយ៉ាងអស់ ពីចិត្ត។ ម៉ាក់ចូលរួមជាមួយនឹងកូនដែរមក។ កូនមិនចង់ឱ្យរឿងបែបនេះ កើតមានបន្តទៅទៀតឡើយ ពីព្រោះកូនចង់ឱ្យប៉ា ទៅព្រះវិហារជាមួយនឹងយើងដែរ»។

ម្តាយ និងកូនស្រី បានទៅព្រះវិហារ នៅក្នុងប្រទេសអាមេនី ដោយគ្មានស្វាមី និងគ្មានឪពុក។ គ្រួសារជាច្រើន បានទទួលភាពស្រណុកស្រួល ជាមួយនឹងការរៀបចំ ដរាបណាឪពុក ឬស្វាមី បានអនុញ្ញាត ឱ្យធ្វើអ្វីមួយ ដោយគ្មានការបៀតបៀន។ ប៉ុន្តែ នាងអានើសលែងសប្បាយចិត្ត ជាមួយនឹងការរៀបចំបែបនេះ ហើយ ដោយនាងចង់ឱ្យឪពុករបស់នាងទៅព្រះវិហារវិញ។

ម្តាយបានយល់ព្រមអធិស្ឋាន។ ក្តីសង្ឃឹមរបស់នាងអានើស បានកើនឡើង។ នាងបានប្រាកដ នៅក្នុងចិត្តថា ព្រះនឹងផ្លាស់ប្តូរចិត្តរបស់ឪពុកនាង ជាមិនខាន។ នាងបានមានជំនឿចិត្តថា រឿងនេះ នឹង អាចកើតឡើងគ្រប់ពេលវេលា។
