### ការនាំជនជាតិអាមេនីឱ្យស្គាល់ព្រះ

ខណៈពេលដែលនាងអានុស កំពុងតែអធិស្ឋាន ឱ្យឪពុករបស់នាង នោះនាងក៏បានអធិស្ឋាន ឱ្យប្រជាជន4000នាក់ នៅក្នុងទីក្រុងរបស់នាង ក្នុងប្រទេសដែរ។ បន្ទាប់មក ព្រះបានចាប់ផ្តើមប៉ះពាល់ ចិត្តប្រជាជនទាំងនោះ។

បន្ទាប់ពីនាងបានជ្រមុជទឹករួច នាងអានុសបានជួបជុំអធិស្ឋានឱ្យទីក្រុងរបស់នាង ជាមួយនឹង ស្ត្រីវ័យក្មេង2នាក់ទៀត។ បន្ទាប់មក ពួកគេបានរៀបចំថ្នាក់សិក្ខាសាលាមួយកន្លែង ដែលបានទាក់ទាញ យុវជនជាច្រើននាក់។ បន្ទាប់ពីនោះមក ពួកគេបានជួយឡានក្រុង និងបានយកយុវជនចំនួន50នាក់ ទៅទស្សនាកម្សាន្ត នៅជុំវិញប្រទេសអាមេនី។ រាល់ពេលដែលឡានក្រុងចតម្តងៗ ក្រុមជំនុំបាននាំគ្នា មកស្វាគមន៍ និងរាក់ទាក់យុវជនទាំងនេះ។ នាងអានុសបានសង្កេតឃើញថា យុវជនទាំងនេះ កាន់តែ ស្មោះត្រង់ និងបើកចំហ ដោយបានសួរណាំអំពីព្រះ នៅពេលដែលពួកគេទៅដើរកម្សាន្ត ឆ្ងាយពីផ្ទះនោះ។

ខណៈពេលដែលគេចាប់អារម្មណ៍សារអាត់វេនទីស្ទ នោះប៉ាស្ទ័រ បានចាប់ផ្តើមចុះមកមើលបញ្ហា ជាក់ស្តែងផ្សេងៗទៀត នៅក្នុងទីក្រុង។ មនុស្សជាច្រើននាក់ បានចូលរួមនៅក្នុងសិក្ខាសាលា ជាង2ឆ្នាំ។ ឪពុករបស់នាងអានុស គឺជាមនុស្សដែលមានឥទ្ធិពល និងមានគេគោរព ដូច្នេះហើយ នៅពេលដែលគាត់ ក្លាយមកជាអ្នកអាត់វេនទីស្ទ នោះអ្នកនៅក្នុងទីក្រុងទាំងអស់ ក៏បានកត់សម្គាល់ដែរ។ ប្រជាជនបាននាំគ្នានិយាយអំពីសេចក្តីជំនឿរបស់គាត់ ព្រមទាំងការសម្រេចចិត្តទៅព្រះវិហារ ដែលមិនមែនជាព្រះវិហារ ថ្នាក់ជាតិ។ ពិធីបុណ្យជ្រមុជទឹករបស់គាត់ បានបំបែកចិត្តរឹងរូស។ ព្រះវិហារអាត់វេនទីស្ទនៅក្នុងទីក្រុង ដែលកាលពីមុន មានត្រឹមតែស្ត្រីស្មោះត្រង់7នាក់ នៅក្នុងព្រះវិហារនៅផ្ទះផ្ទាល់ខ្លួននោះ ឥឡូវនេះ បាន ផ្លាស់ទៅបន្ទប់ជួល ដែលមានមនុស្សរាប់សិបនាក់ បានរួបរួមគ្នាថ្វាយបង្គំជារៀងរាល់ថ្ងៃសប្បាតហ៍។ ក្រុម ជំនុំ និងអ្នកដទៃទៀត ក៏បានជួបជុំគ្នា អធិស្ឋាននៅតាមបណ្តាញអនឡាញជាប្រចាំផងដែរ។ គម្រោង នៃការសាងសង់អគារថ្មី ក៏កំពុងតែដំណើរការទៅដែរ។

សព្វថ្ងៃនេះ ឪពុក ដែលមានឈ្មោះថា អារម៉ែន សាហ្វាយ៉ាន់ (Armen Safaryan) បានចូលរួម ធ្វើការ ជាមួយភរិយាឈ្មោះថា ហ្គាយ៉ាន់ បាដាលីយ៉ាន់ និងកូនស្រីឈ្មោះថា អានុស សាហ្វាយ៉ាន់ ដើម្បី ធ្វើតៅហ៊ូ ចំនួន3ប្រភេទ នៅក្នុងក្រុមហ៊ុនរបស់ពួកគេ។ នៅក្នុងនាមជាក្រុមហ៊ុនលក់តៅហ៊ូតែមួយគត់ នៅក្នុងប្រទេសអាមេនី ក្រុមហ៊ុននេះ ត្រូវបានចាក់ផ្សាយ នៅតាមកញ្ចក់ទូរទស្សន៍ជាតិ ហើយបានផ្តល់ ឱកាស ឱ្យឪពុកបានចែករំលែកសេចក្តីជំនឿរបស់គាត់ នៅពេលដែលគេសួរគាត់ថា ហេតុអ្វីបានជាគាត់ នឹកចង់បង្កើតតៅហ៊ូយ៉ាងដូច្នេះ។

ឪពុក គឺជាចាស់ទុំ នៅក្នុងព្រះវិហារ និងជាប្រធាននៃនាយកដ្ឋានគ្រួសារ ហើយគាត់ និងម្តាយ ដែលកាន់នាយកដ្ឋានសុខភាព នៅក្នុងព្រះវិហារ ក៏មានទំនួលខុសត្រូវខ្ពស់ នៅតាមព្រះវិហារដទៃទៀត ដែរ។ ពីព្រោះអ្នកផងបានចាត់ទុកឪពុកថា ជាគំរូមួយ នៅក្នុងជនបទ ដែលមានម្តាយ និងកូនជាច្រើននាក់ បានទៅព្រះវិហារ ដោយគ្មានស្វាមី និងគ្មានឪពុក។ ដូច្នេះហើយ បានជាឪពុក ម្តាយ និងនាងអានុស ចង់ ផ្លាស់ប្តូរចំណុចនេះ។ អ្នកដឹកនាំព្រះវិហារ បានណែនាំឪពុក ឱ្យសមាជិកគ្រប់គ្នាបានស្គាល់ យ៉ាងដូច្នេះ ថា «សូមអ្នកទាំងអស់គ្នាមើលមកនេះ បងប្រុស ដែលជាជនជាតិអាមេនី ដ៏សាមញ្ញម្នាក់នេះ គឺជាអ្នក អាត់វេនទីស្ទ ដូច្នេះហើយ បងប្អូនទាំងអស់គ្នា មិននៅតែលតោលនោះទេ ពីព្រោះបងប្រុសម្នាក់នេះ នឹង មកព្រះវិហារនេះ ជារៀងរាល់ថ្ងៃសប្បាតហ៍»។

នាងអានុស បានចែករំលែករឿងរបស់នាង នៅតាមព្រះវិហារ និងតាមជំរំយុវជនយ៉ាងដូច្នេះ ថា «សូមកុំឱ្យស្កប់ចិត្តជាមួយនឹងស្វាមី និងឪពុករបស់លោកអ្នក ដោយគ្រាន់តែគាត់បានអនុញ្ញាត ឱ្យអ្នក មកព្រះវិហារប៉ុណ្ណោះទេ ប៉ុន្តែ ចូរទូលសូមជាមួយនឹងព្រះ ដើម្បីឱ្យពួកគេ បានទៅព្រះវិហារ ជាមួយនឹង លោកអ្នកដែរ»។
