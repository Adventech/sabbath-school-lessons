### ជម្រើសអាហារ ដែលបានបង្កឱ្យមានភាពចលាចល

នៅថ្ងៃសប្បាតហ៍មួយ នាងអានុស (Anush) និងភរិយារបស់គាត់ បានត្រឡប់ពីព្រះវិហារ មកវិញ ហើយបានដឹងថា ឪពុកបានរៀបចំគម្រោង ទៅញុាំអាហារកម្សាន្ត នៅតាមជនបទ។ គាត់បាន និយាយថា «តោះយើង ញុាំសាច់អាំង»។ លោកអេនុសបាននឹកចាំ ពីរបៀបដែលជនជាតិអ៊ីស្រាអែល បានរៀបចំម្ហូបអាហារថ្ងៃសប្បាតហ៍ របស់ពួកគេ នៅថ្ងៃសុក្រ មុនពេលដែលថ្ងៃសប្បាតហ៍ចូលមកដល់ (និក្ខមនំ 16) ប៉ុន្តែ គាត់ក៏នឹកឆ្ងល់ នៅក្នុងចិត្តដែរថា តើការទៅអាំងសាច់នៅថ្ងៃសប្បាតហ៍ ជាគំនិតល្អ ដែរឬទេ។ នាងអានុសបានឆ្លើយយ៉ាងមឺងម៉ាត់ថា «យើងមិនគួរចេញទៅអាំងសាច់នៅថ្ងៃសប្បាតហ៍នោះទេ ហើយនេះក៏ជាផែនការល្អដែរ ពីព្រោះខ្ញុំអត់ញុាំសាច់ផងហ្នឹង»។ នាងបានក្លាយជាអ្នកញុាំអាហារបួស។ ឪពុកបានរំសាយគម្រោងញុាំអាហារកម្សាន្ត ប៉ុន្តែ គាត់នៅតែមិនយល់ថា ហេតុអ្វីបានជានាងអានុស លែងញុាំសាច់យ៉ាងដូច្នេះ។ នៅថ្ងៃបន្ទាប់ គាត់បានចម្អិនមាន់មួយ សម្រាប់អាហារពេលថ្ងៃ រួចហែកសាច់ មាន់នោះ 1ចំណិតឱ្យនាង។ នាងអានុសបាននិយាយថា «ប៉ា ខ្ញុំអត់ញុាំសាច់ទេ»។ ឥឡូវនេះ ឪពុកបាន យល់ពីស្ថានភាពរបស់នាង ដែលធ្វើឱ្យគាត់មិនសប្បាយចិត្ត។ គាត់បានយល់ឃើញថា ការមិនញុាំសាច់ គឺជារឿងមិនប្រក្រតី។ នៅថ្ងៃបន្ទាប់ គាត់បានហាមមិនឱ្យនាងអានុស និងម្តាយ ទៅចូលរួមអធិស្ឋាន នៅក្នុងព្រះវិហារនៅតាមផ្ទះ ក្នុងទីក្រុងរបស់ពួកគេនោះឡើយ។ នៅពេល ដែលនាង និងម្តាយប្រកែក ឪពុកបានបញ្ចេញកំហឹង ដែលគាត់បានលាក់ទុក ទាស់នឹងអ្នកអាត់វេនទីស្ទ។ គាត់បានរិះគន់ទៅនឹង សំណូមពរ ដែលព្រះគម្ពីរបានប្រាប់ ឱ្យបង្វិលតង្វាយ10ហូត1 និងតង្វាយសទ្ធា (ម៉ាឡាគី 3:8-10)។ គាត់បាននិយាយថា «តង្វាយ10ហូត1 និងតង្វាយសទ្ធា គឺជាជំនួញ ដូច្នេះហើយ កូនគ្រាន់តែជាអ្នកជួយ គាំទ្រជំនួញគេប៉ុណ្ណោះ»។

ឪពុកបានចោទប្រកាន់ព្រះវិហារអាត់វេនទីស្ទថា ជាក្រុមបរទេស ដែលមានបំណងចង់បំផ្លាញ ប្រទេសអាមេនី។ គាត់បានរិះគន់របៀបរស់នៅរបស់នាងអានុស។ គាត់បាននិយាយថា «ថ្ងៃនេះ កូន និយាយថា កូនមិនញុាំសាច់ នៅថ្ងៃស្អែក កូននឹងនិយាយទៀតថា «កូនមិនមានឪពុកនោះទេ»។ នាងអានុស បានអង្គុយអធិស្ឋានយ៉ាងស្ងៀមស្ងាត់ថា «ឱព្រះអម្ចាស់អើយ តើគួរឱ្យទូលបង្គំនិយាយយ៉ាងណា?»។ រាល់ពេលដែលឪពុករបស់នាង បាននិយាយទាស់នឹងព្រះ ឬក្រុមជំនុំ នាងតែងតែអធិស្ឋានថា «ប៉ាមិនមែន និយាយមកកាន់ខ្ញុំទេ តែគាត់បាននិយាយទៅកាន់ទ្រង់វិញ ដូច្នេះហើយ ទ្រង់គឺជាអ្នកទទួលខុសត្រូវ នៅក្នុង ការឆ្លើយតបនឹងគាត់វិញ»។ នាងបាននឹកចាំគម្ពីររ៉ូម 2:4 ដែលបានចែងថា «តើអ្នកមិនដឹងថា សេចក្តីសប្បុរសរបស់ព្រះ នាំអ្នកឲ្យប្រែចិត្តទេឬ?»។ នាងហាក់ដូចជាមានអារម្មណ៍ថា ព្រះកំពុងតែមានបន្ទូល ដើម្បីពង្រីកភាពសប្បុរសស្រដៀងគ្នានេះ នៅក្នុងឪពុករបស់នាង។ នាងបានអធិស្ឋានទៀតថា «ទូលបង្គំ មិនអាចធ្វើអ្វីបានទេ ក្រៅពីស្រឡាញ់ឪពុករបស់ទូលបង្គំនោះ»។

ឪពុកមានហាងលក់គ្រឿងទេសតូចមួយ។ ជារៀងរាល់ពេលដែលគាត់ទុកឱ្យភរិយារបស់គាត់ ឬនាងអានុសឱ្យមើលការខុសត្រូវ ពួកគេមិនដែលលក់ស្រា ឬបារីម្តងណាឡើយ។ ឥឡូវនេះ គាត់មាន អារម្មណ៍ថា ព្រះបានថ្កោលទោសគាត់វិញ នៅពេលដែលគាត់បានស្តីបន្ទោសឱ្យម្តាយ និងកូនស្រីរបស់ គាត់។ គាត់បានសួរនាងអានុសថា «តើកូនគិតថា ប៉ាជាមនុស្សអាក្រក់ ហើយកូនជាមនុស្សល្អ ដោយសារ តែប៉ាបានលក់ស្រា និងបារី តែកូនមិនបានលក់ មែនទេ? ចាប់ពីពេលនេះតទៅ ប៉ានឹងធ្វើជាគ្រីស្ទានឱ្យ ល្អជាងកូនទៅទៀត ប៉ានឹងដឹកនាំកម្មវិធីថ្វាយបង្គំព្រះនៅថ្ងៃសប្បាតហ៍ កូន និងម៉ាក់មិនបាច់ទៅព្រះវិហារ ទៀតឡើយ»។

ពាក្យសម្តីរបស់ឪពុក បានបញ្ចប់នូវការសន្ទនានេះ។ នាងអានុសបានទៅបន្ទប់របស់នាង ហើយ ម្តាយក៏បានទៅតាមនាងដែរ។ ទាំងពីរនាក់មានការភ្ញាក់ផ្អើលជាខ្លាំង។ ម្តាយបានសួរកូនថា«តើយើង គួរតែធ្វើយ៉ាងណាទៀត?»។ នាងអានុសបានឱ្យយោបល់ថា ពួកគេគួរតែសហការ ជាមួយឪពុកតទៅ ទៀត បើសិនជាគាត់មិនបានប្រឆាំងនឹងព្រះគម្ពីរនោះ។ នាងបាននិយាយថា «ប៉ាគាត់បាននិយាយថា យើងនឹងថ្វាយបង្គំព្រះនៅក្នុងផ្ទះរបស់យើង ពីព្រោះគាត់មិនបានដកព្រះគម្ពីរពីយើងឡើយ គាត់ថែមទាំង គោរពថ្ងៃសប្បាតហ៍ទៀតផង ដូច្នេះហើយ យើងគួរតែរង់ចាំឱ្យថ្ងៃសប្បាតហ៍មកដល់សិន បើសិនជាគាត់ គោរពសម្តីរបស់គាត់ពិតមែន នោះយើងនឹងរក្សាថ្ងៃសប្បាតហ៍ នៅក្នុងផ្ទះជាមួយនឹងគាត់វិញ តែបើសិនជា គាត់មិនបានគោរពសម្តីរបស់គាត់វិញ នោះយើងនឹងអធិស្ឋាន ហើយសង្កេតមើលពីរបៀបដែលព្រះបាន ដឹកនាំយើងវិញ»។
