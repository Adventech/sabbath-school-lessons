### ការផ្តល់ក្តីសង្ឃឹមដល់ប្រជាជនម៉ាសៃ

នៅក្នុងប្រទេសកេនយ៉ា លោករ៉ូមផាស ដែលក្មេងប្រុសជនជាតិម៉ាសៃអាយុ១៦ឆ្នាំ បានសម្រេច ចិត្តទៅសាលារៀន បន្ទាប់ពីគាត់បានទទួលបុណ្យជ្រមុជទឹក។ គាត់ចង់អានព្រះគម្ពីរដោយខ្លួនឯង លើសពី រឿងផ្សេងៗទៀត។

លោករ៉ូមផាសបានប្រឈមមុខនឹងការជំទាស់ ពីឪពុករបស់គាត់ និងពីបងប្អូនចំនួន៨២នាក់ របស់គាត់ ដែលគិតថា ក្មេងប្រុសនៃជនជាតិម៉ាសៃ មិនចាំបាច់ទទួលបានការអប់រំនោះទេ។ ប៉ុន្តែ លោក រ៉ូមផាសបានក្លាយជាមនុស្សដំបូងគេ នៅក្នុងគ្រួសាររបស់គាត់ ដែលបានបញ្ចប់ថ្នាក់ទី១២ និងបញ្ចប់ថ្នាក់ វិទ្យាល័យ។ បន្ទាប់ពីនោះ លោករ៉ូមផាសបានសម្រេចចិត្តសិក្សាព្រះគម្ពីរ នៅឯប៊ូហ្គីម៉ា (Bugema) ដែលជា សាកលវិទ្យាល័យសេវេនដេយអាត់វេនទីស្ទ នៅក្នុងប្រទេសអ៊ូហ្គង់ដា (Uganda)។ ប៉ុន្តែ លោករ៉ូមផាស ត្រូវការថវិកា។

នៅល្ងាចមួយ លោករ៉ូមផាសបានហៅបងប្អូនរបស់គាត់ ដែលដូចគាត់ ឱ្យចូលរួមថ្វាយបង្គំនៅថ្ងៃសប្បាតហ៍ និងបានសុំឱ្យពួកគេអធិស្ឋានឱ្យគាត់ ដើម្បីឱ្យគាត់រកបានថវិកា៧,០០០ស៊ីលីងកេនយ៉ា (២១៨,៨៤៤.៩៦ រៀល) យកមកធ្វើដំណើរទៅកាន់ប្រទេសអ៊ូហ្គង់ដា ដើម្បីដាក់ពាក្យសុំចូលរៀននៅឯ សាកលវិទ្យាល័យប៊ូហ្គីម៉ា។ បងប្អូនរបស់គាត់បានអធិស្ឋាន ពេលដែលលោករ៉ូមផាសក្រាបនៅលើឥដ្ឋ។ បន្ទាប់ពីបាននិយាយពាក្យចុងក្រោយថា «អាម៉ែន» ស្រាប់តែមានសំឡេងគោះទ្វារបានបន្លឺឡើង។ អ្នកគោះទ្វារគឺជាអ្នកនយោបាយម្នាក់ឈ្មោះអាឡិច (Alex) ដែលបានមកជួបឪពុករបស់លោករ៉ូមផាស។ អ្នកនយោបាយចូលចិត្តមកលេងផ្ទះគាត់ ពីព្រោះនៅពេលបោះឆ្នោត គ្រួសារដ៏ធំរបស់រ៉ូមផាស តំណាង ឱ្យសំឡេងឆ្នោតនៃមនុស្សជាច្រើន។ អ្នកនយោបាយម្នាក់នេះ មិនមែនជាអាត់វេនទីស្ទនោះទេ ដូច្នេះហើយ គាត់បានសួរសំណួរមួយយ៉ាងសាមញ្ញថា៖ «តើគ្រួសារធំមួយ មានប៉ាស្ទ័រដែរឬទេ?»។ លោករ៉ូមផាស បានណែនាំខ្លួន ដល់អ្នកនយោបាយម្នាក់នេះថា តាំងពីក្មេងមក បងប្អូនរបស់គាត់បានហៅគាត់ថា «ប៉ាស្ទ័រ»។ លោកអាឡិចបានសួរទៀតថា «តើក្មួយប្រុសត្រូវការអ្វីជាងគេបំផុត?»។ លោករ៉ូមផាសបានឆ្លើយថា «ខ្ញុំចង់ទទួលបានសញ្ញាបត្រពីសាកលវិទ្យាល័យប៊ូហ្គីម៉ា នៅឯប្រទេសអ៊ូហ្គង់ដា»។ លោកអាឡិចបានដកយក១៥,០០០ស៊ីលីងកេនយ៉ា (៤៦៨,៩៥៣.៤៩) ចេញពីហោប៉ៅរបស់គាត់ ឱ្យទៅលោក រ៉ូមផាស។ ប្រាក់ដែលលោករ៉ូមផាសទទួលបានពីអ្នកនយោបាយ មានចំនួនជាងទ្វេដង លើសពីការអធិស្ឋាន របស់គាត់។ លោករ៉ូមផាសបានធ្វើដំណើរទៅកាន់ប្រទេសអ៊ូហ្គង់ដា ហើយបានចុះចូលរៀននៅក្នុងកម្មវិធីសិក្សាព្រះគម្ពីរ។ បន្ទាប់មក លោករ៉ូមផាសបានត្រឡប់មកផ្ទះវិញ ដើម្បីរង់ចាំថ្ងៃចូលរៀន។ នៅថ្ងៃដដែល ដែលគាត់បានមកដល់ផ្ទះនោះ លោកអាឡិចបានចុះមកសួរសុខទុក្ខម្តងទៀត។ នៅពេល ដែលគាត់ឮថា សាលាបានទទួលយកលោករ៉ូមផាស នោះគាត់បានឱ្យលុយដុល្លារដល់លោករ៉ូមផាស។ លោករ៉ូមផាសមិនធ្លាប់កាន់លុយដុល្លារ ពីមុនមកនោះទេ។ ប្រាក់នោះល្មមគ្រប់គ្រាន់សម្រាប់បង់សាលា នៅក្នុងរយៈពេល៣ឆ្នាំ។

សព្វថ្ងៃនេះ លោករ៉ូមផាស ចូស្វាត ឡេគីសុន (Rompas Josphat Lekishon) គឺជាប៉ាស្ទ័រ នៃព្រះវិហារអាត់វេនទីស្ទ ដែលមានចិត្តផុសផុលនៅក្នុងបេសកកម្ម។ តាមរយៈកិច្ចខិតខំប្រឹងប្រែងរបស់ គាត់ ព្រះវិហារចំនួន៦កន្លែង បានក្លាយមកជាព្រះវិហារសេវេនដេយអាត់វេនទីស្ទ។ លោករ៉ូមផាសក៏បានបើកព្រះវិហារមួយកន្លែងទៀត នៅលើដីនៃឪពុករបស់គាត់ ដែលឪពុកបានបរិច្ចាគឱ្យព្រះវិហារអាត់វេនទីស្ទ។ សមាជិក៣៣គ្រួសារ បានថ្វាយបង្គំនៅទីនោះ ជារៀងរាល់ថ្ងៃសប្បាតហ៍។ លោក រ៉ូមផាស ចូលចិត្តចែករំលែកដំណឹងល្អអំពីព្រះយេស៊ូវ ជាមួយនឹងប្រជាជនម៉ាសៃ និងបានចែកចាយព្រះគម្ពីរជាភាសាម៉ាសៃ ជាង៥០០ក្បាល។ លោករ៉ូមផាសបាននិយាយថា «រឿងដែលខ្ញុំស្រឡាញ់បំផុតនោះ គឺជាការចែកចាយព្រះគម្ពីរដល់ប្រជាជនម៉ាសៃនេះឯង ពីព្រោះការបែបនេះ នឹងផ្តល់ក្តីសង្ឃឹមដល់អ្នកដែលគ្មានទីពឹង»។
