### ការបង្កើត ទំនាក់ទំនង ជាមួយនឹង ព្រះ

លោកអ្នកប្រហែលជាបានធំធាត់ឡើង នៅក្នុងនាមជាសេវេនដេយអាត់វេនទីស្ទ ឬជាដែលមានជំនឿ ថ្មី។ លោកអ្នកប្រហែលជាបានរៀនមគ្គុទេសក៍សិក្សាព្រះគម្ពីរសម្រាប់មនុស្សចាស់ ឬបានអានមតិមួយ ចំនួនពីសៀវភៅនេះ។ មិនថា រឿងរបស់លោកអ្នកបែបណានោះទេ ប្រធានអំពីរបៀបបង្កើនទំនាក់ទំនង ជាមួយនឹងព្រះ គឺជាប្រធានបទសម្រាប់លោកអ្នក។​

ប្រធានបទនេះ បានជួយឱ្យយើងយល់កាន់តែច្បាស់ អំពីប្រធានបទដទៃទៀតនៅក្នុងព្រះគម្ពីរ។ តើសព្វថ្ងៃនេះ លោកអ្នកគិតថា ព្រះមានរូបរាងបែបណា? តើលោកអ្នកបានស្គាល់ និងបានស្រឡាញ់ ទ្រង់ដែរឬទេ ឬមួយក៏ទ្រង់ជាជនចម្លែកសម្រាប់លោកអ្នក? បើសិនជាទ្រង់ជាជនចម្លែក នោះចូរសូមទូល ឱ្យព្រះជួយឱ្យលោកអ្នកបានយល់អំពីទ្រង់កាន់តែខ្លាំងឡើង ខណៈពេលដែលលោកអ្នកសិក្សាមេរៀនសប្បាតហ៍ស្គូល។ លោកអ្នកប្រហែលជានឹកឆ្ងល់ថា តើលោកអ្នកត្រូវពេលវេលានៃការអធិស្ឋាន និងការសិក្សាព្រះគម្ពីររបស់លោកអ្នក ឬលោកអ្នកត្រូវប្រឆាំងទាស់នឹងអំនួតនៅក្នុងចិត្តដោយរបៀបណា។ លោកអ្នក អាចនឹងប្រាថ្នាចង់បានជំនឿ និងចំណេះដឹងបន្ថែមទៀត ឬលោកអ្នកចង់យល់ឱ្យកាន់តែច្បាស់ឡើង អំពី ទំនាក់ទំនងរវាងអំពើបាប និងបញ្ញត្តិរបស់ព្រះ ឬលោកអ្នកចង់រៀនបន្ថែមទៀត អំពីប្រធានបទនៃការ លើកលែងទោស ឬលោកអ្នកចង់រៀនអំពីរបៀបយកឈ្នះសង្គ្រាមទាស់នឹងអំពើបាប និងការខកចិត្ត ឬមួយ ក៏លោកអ្នកអាចនឹងប្រាថ្នាចង់រៀនអំពីរបៀបកម្សាន្តចិត្តអ្នកដទៃ នៅក្នុងអំឡុងពេលដែលពួកគេបាន ជួបនឹងគ្រាលំបាក។

ទំនាក់ទំនងរបស់លោកជាមួយនឹងព្រះ គឺជាទំនាក់ទំនងសំខាន់បំផុតរបស់លោកអ្នក។ សូមកុំរង់ចាំ បង្កើតវាអី ប៉ុន្តែ សូមឱ្យទំនាក់ទំនងនេះបានរឹងមាំឡើង តាមដែលលោកអ្នកអាចធ្វើទៅបាន។ ឥឡូវនេះ ដល់ពេលពេលដែលលោកអ្នកដំណើរការទំនាក់ទំនងនេះហើយ។ លោកអ្នកមិនត្រូវរង់ចាំពេលណាមួយ នោះទេ។ ទំនាក់ទំនងនេះ នឹងកែលម្អទំនាក់ទំនងដទៃទៀតដែលយើងមាន។ ទំនាក់ទំនងដែលលោកអ្នក មានជាមួយនឹងព្រះ នឹងជួយកែលម្អទំនាក់ទងរបស់លោកអ្នក ជាមួយនឹងកូនៗ (ប្រសិនបើលោកអ្នក មានកូន)។ ទំនាក់ទំនងរបស់លោកអ្នកជាមួយនឹងព្រះ ក៏នឹងជួយឱ្យលោកអ្នក សម្រេចចិត្តបានត្រឹមត្រូវ អំពីការប្រើប្រាស់ថវិក អំពីរឿងដែលលោកអ្នកធ្វើ ដើម្បីទទួលបានភាពរីករាយ និងអំពីសេចក្តីសង្ឃឹម និង សុបិនរបស់លោកអ្នក។ ជាការពិតណាស់ ទំនាក់ទំនងរបស់លោកអ្នកជាមួយនឹងព្រះ ក៏នឹងផ្តល់ក្តីសង្ឃឹម ដល់លោកអ្នក សម្រាប់ជីវិតនៅពេលអនាគត ជាមួយនឹងព្រះយេស៊ូវផងដែរ។

ព្រះទ្រង់ចង់មានទំនាក់ទំនងជាមួយនឹងលោកអ្នក គឺជាប្រធានបទធំធេង ដែលមានពេញនៅក្នុងព្រះគម្ពីរ

ដូច្នេះហើយ យើងអាចសិក្សាប្រធានបទនេះ ទៅតាមវិធីជាច្រើនផ្សេងៗគ្នា។ យើងអាចអានរឿង ទស្សនៈ និងខគម្ពីរអំពីប្រធាននេះ។ ជាការពិតណាស់ សប្តាហ៍នេះ យើងអាចនឹងសិក្សាត្រឹមតែគោលគំនិត និងខគម្ពីរមួយចំនួនប៉ុណ្ណោះ។

មិនថា សព្វថ្ងៃនេះ លោកអ្នកមានទំនាក់ទំនងជាមួយនឹងព្រះបែបណានោះទេ មេរៀនទាំងនេះត្រូវបាន ចងក្រងឡើងសម្រាប់លោកអ្នក។ នៅទីបញ្ចប់ មេរៀនទាំង១៣នេះ អាចនឹង «ដាស់» សេចក្តីស្រឡាញ់ ដែលលោកអ្នកមានចំពោះព្រះយេស៊ូវ។​ មេរៀនទាំងនេះ អាចនឹងជួយឱ្យលោកអ្នកស្វែងរកព្រះយេស៊ូវ ព្រមទាំងជួយឱ្យលោកអ្នកថ្វាយជីវិតរបស់លោកអ្នកដល់ទ្រង់ ឱ្យបានពេញលេញបន្ថែមទៀត។

ដោយសារតែមគ្គុទេសកន៍ស្តីអំពីទំនាក់ទំនងនេះឯង បានជាមេរៀនទាំងនេះ មានលក្ខណៈខុសគ្នា ពីមេរៀនកាលពីមុនបន្តិចបន្តួច។ មេរៀនទាំងនេះ គឺជាមេរៀនផ្ទាល់ខ្លួន ពីព្រោះមេរៀនទាំងនេះ បាន ស្តីអំពីព្រះផ្ទាល់ខ្លួន។ ព្រះផ្ទាល់ខ្លួននេះ មានបំណងចង់ស្គាល់លោកអ្នកដោយផ្ទាល់។

អ្នកស្រីអេឡែន ជីវ៉ៃត៍បានមានប្រសាសន៍ថា៖ «ជីវិតនៃគ្រីស្ទានដែលតែងតែស្មោះត្រង់ចំពោះព្រះ គឺជាបាដិហារ្យដ៏ធំអស្ចារ្យបំផុត»។ ដកស្រង់ពី The Desire of Ages, page 407, adapted។ ព្រះគម្ពីរបានប្រើរូបស័ព្ទនៃការរត់ប្រណាំង ដើម្បីបកស្រាយអំពីទំនាក់ទំនងរបស់យើងជាមួយនឹងព្រះ។ ការប្រណាំងភាគច្រើន បានផ្តល់រង្វាន់ ឬជ័យលាភីដល់អ្នកឈ្នះ។ ដូចគ្នានេះ រង្វាន់របស់យើងគឺជាម្កុដ (កូរិនថូសទី១ ៩:២៤, ២៥) និងជីវិតអស់កល្បជានិច្ចជាមួយនឹងព្រះ។ «ការរត់ប្រណាំង» ខាងឯវិញ្ញាណ របស់យើង គឺជាការរត់យកចម្ងាយ មិនមែនជាការរត់តែមួយភ្លេតនោះទេ។ មានពេលខ្លះ យើងអាចនឹង ឈប់រត់ ឬមានពេលខ្លះ យើងអាចនឹងជំពប់ដួល។ ប៉ុន្តែ នៅពេលដែលរឿងនេះកើតឡើង នោះយើង គ្រាន់តែក្រោកឡើង ហើយរត់បន្តទៀតប៉ុណ្ណោះ។ យើងត្រូវតែបន្ត «រត់ប្រណាំង» បើទោះបីជាយើងបាន រងនូវការលំបាកក៏ដោយ។ ជាការពិតណាស់ បញ្ហានឹងត្រូវមកដល់ (ហេព្រើរ ១២:៤-១១)។ ប៉ុន្តែ អ្វីដែលជា ដំណឹងល្អនោះ គឺយើងមិនបាន «រត់ប្រណាំង» តែម្នាក់ឯងនោះទេ។ មានអ្នកដទៃទៀត បានរត់ប្រណាំង ជាមួយនឹងយើង។​ អ្នករត់ប្រណាំងទាំងនេះ គឺជាអ្នកដែលស្រឡាញ់ព្រះ និងព្រះគម្ពីរ។ អ្វីដែលសំខាន់ បំផុតនោះ គឺព្រះយេស៊ូវបានប្រទានអង្គជំនួយដល់យើង។​ អង្គជំនួយ គឺជាព្រះវិញ្ញាណបរិសុទ្ធ។ ព្រះយេស៊ូវ បានសន្យាថា «នោះ​ខ្ញុំ​នឹង​ទូល​សូម​ដល់​ព្រះវរ‌បិតា ហើយ​ទ្រង់​នឹង​ប្រទាន​ព្រះ​ដ៏​ជា​ជំនួយ​១​អង្គ​ទៀត មក ​អ្នក​រាល់​គ្នា ឲ្យ​បាន​គង់​នៅ​ជា​មួយ នៅ​អស់‌កល្ប​រៀង​ទៅ គឺ​ជា​ព្រះ‌វិញ្ញាណ​នៃ​សេចក្ដី​ពិត ដែល​លោកីយ៍ ទទួល​ពុំ​បាន ព្រោះ​មិន​ឃើញ ហើយ​មិន​ស្គាល់​ទ្រង់​សោះ តែ​អ្នក​រាល់​គ្នា​ស្គាល់​ទ្រង់ ដ្បិត​ទ្រង់​គង់​ជា​មួយ ក៏​នឹង​សណ្ឋិត​នៅ​ក្នុង​អ្នក​រាល់​គ្នា​ដែរ» (យ៉ូហាន ១៤:១៦, ១៧)។

សូមសរសើរតម្កើងព្រះ ដែលយើងមិនបាន «រត់ប្រណាំង» ខាងឯជីវិតតែម្នាក់ឯង។ អង្គជំនួយ​ បានគង់នៅជាមួយនឹងយើង។ ព្រះអង្គក៏បានសណ្ឋិតនៅក្នុងយើងដែរ។ ព្រះបានធ្វើឱ្យយើងរឹងមាំ។ ព្រះអង្គ បានជួយដល់យើង ខណៈពេលយើងរត់ប្រណាំង។ ព្រះអង្គបានជួយឱ្យយើង «​រំពឹង​មើល​ដល់​ព្រះ‌យេស៊ូវ​ ដ៏​ជា​មេ​ផ្តើម ហើយ​ជា​មេ​សំរេច​សេចក្ដី​ជំនឿ​របស់​យើង» (ហេព្រើរ ១២:២)។

ខណៈពេលដែលខ្ញុំ (នីណា អេធចេសុន) សរសេរមគ្គុទេសកន៍នេះ នោះខ្ញុំបានអធិស្ឋានទូលសូមឱ្យព្រះវិញ្ញាណបរិសុទ្ធប៉ះពាល់ចិត្តរបស់យើង ដែលជាគ្រីស្ទាន និងជាក្រុមជំនុទូទៅពិភពលោក។ សូម ឱ្យយើងបានជាប់ជាមួយនឹងព្រះ។ ទំនាក់ទំនងយើងជាមួយនឹងទ្រង់ គឺជារឿងសំខាន់បំផុតនៅក្នុងការរស់នៅរបស់យើង។

ដូច្នេះហើយ ចូរយើងសិក្សាមេរៀននេះទាំងអស់គ្នា។ ក៏សូមឱ្យយើងទាំងអស់គ្នា បានសិក្សា ស្រឡាញ់ និងរស់នៅក្នុងព្រះផងដែរ។
