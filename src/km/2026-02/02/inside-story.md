### ការបដិសេធមិនហោះហើរ នៅថ្ងៃសប្បាតហ៍

លោកខាសេរេខា ម៉ាចហូលូ ខាងវ៉ាត់ស៊ី (Kasereka Maghulu Kavatsi) ដែលជាអ្នកជំនួញ នៃព្រះវិហារសេវេនដេយអាត់វេនទីស្ទ បានសង់អាណាចក្រពាណិជ្ជកម្ម នៅក្នុងសាធារដ្ឋប្រជាធិបតេយ្យ កុងហ្គោ ដែលបានរួមបញ្ចូលទាំងកសិកម្ម ការនេសាទ សម្លៀកបំពាក់ និងការដឹកជញ្ជូនទំនិញតាមក្រុមហ៊ុន អាកាសចរណ៍ខាវ៉ាត់ស៊ី (Kavatsi)។

មានពេលមួយ ក្រុមហ៊ុនអាកាសចរណ៍ខាវ៉ាត់ស៊ី បានជួលយន្តហោះ បូអ៊ីង ៧០៧ (Boeing 707) ពីដៃគូនៅឯប្រទេសកេនយ៉ា។ វាជាការព្រមព្រៀងដ៏ពិបាក នៅក្នុងការសហការជាមួយនឹងដៃគូពាណិជ្ចកម្ម នៅក្នុងការជួលក្រុមហ៊ុនឌូបៃ (Dubai) ដែលក្រុមហ៊ុននេះ បានបន្តជួលយន្តហោះពីម្ចាស់ ក្រុមហ៊ុននៅឯប្រទេសអង់គ្លេស។ នៅក្រោមលក្ខខណ្ឌនៃការជួល ក្រុមហ៊ុនអាកាសចរណ៍ត្រូវចែករំលែក ប្រាក់ចំណេញរបស់ខ្លួន ជាមួយក្រុមហ៊ុននៅឯប្រទេសកេនយ៉ា ប្រទេសឌូបៃ និងប្រទេសអង់គ្លេស។ ការដឹកជញ្ជូនទំនិញបានដំណើរការរលូន រហូតថ្ងៃសុក្រមួយ អាកាសចរណ៍ខាវ៉ាត់ស៊ីចាំបាច់ត្រូវតែផ្អាកសិន ដោយសារតែអាកាសធាតុមិនល្អ។ អាកាសចរណ៍កេនយ៉ា ដែលជាដៃគូ បានទទូចឱ្យយន្តហោះធ្វើដំណើរ នៅថ្ងៃសៅរ៍វិញ ប៉ុន្តែ លោកខាវ៉ាត់ស៊ីបានបដិសេធសំណើនេះ។ លោកខាសេរេខាបាននិយាយថា «ខ្ញុំមិនអាចហោះហើរនៅថ្ងៃសៅរ៍បានទេ»។

ក្រុមហ៊ុនកេនយ៉ា បានទូរស័ព្ទទៅកាន់ក្រុមហ៊ុនឌូបៃថា «យើងបានផ្តល់យន្តហោះរបស់យើង ដល់បុរសនេះ ប៉ុន្តែ លោកខាសេរេខាមិនចង់ហោះនៅថ្ងៃសៅរ៍ទេ។ យើងបានព្យាយាមបញ្ចុះបញ្ចូលគាត់ ថា នេះគឺជារឿងអាជីវកម្ម»។ ក្រុមហ៊ុនឌូបៃបានស្នើសុំទៅកាន់លោកខាសេរេខា ប៉ុន្តែ គាត់មិនព្រមហោះហើរ ឡើយ។ ដូច្នេះហើយ ក្រុមហ៊ុនឌូបៃបានទាក់ទងទៅកាន់ម្ចាស់ក្រុមហ៊ុនធំ នៅឯប្រទេសអង់គ្លេស ដើម្បីបញ្ជា ទៅកាន់ដៃគូកេនយ៉ា ឱ្យដកយន្តហោះចេញពីលោកខាវ៉ាត់ស៊ី ហើយឈប់រកស៊ីជាមួយនឹងក្រុមហ៊ុនអាកាសចរណ៍ខាវ៉ាត់ស៊ីតទៅទៀត។ ម្ចាស់ក្រុមហ៊ុនបានទូរស័ព្ទទៅកាន់លោកខាសេរេខា ហើយបានសួរ គាត់ថា «ហេតុអ្វីបានជាអ្នកមិនព្រមហោះហើរនៅថ្ងៃនេះ?»។ លោកខាសេរេខាបាននិយាយ «ខ្ញុំជា សេវេនដេយអាត់វេនទីស្ទ»។ «តើសេវេនដេយអាត់វេនទីស្ទនេះ ជាព្រះវិហារមែនទេ?»។ លោកខាសេរេខា បានឆ្លើយថា «បាទលោក នេះគឺជាឈ្មោះព្រះវិហារខ្ញុំ»។ បន្ទាប់មក ម្ចាស់ក្រុមហ៊ុនធំបានទាក់ទងទៅកាន់ ដៃគូផ្សេងទៀត ដែលបានជួលយន្តហោះឱ្យ អាកាសចរណ៍ខាវ៉ាត់ស៊ី ដើម្បីសួរនាំអំពីលោកខាសេរេខា។ ពួកគេបាននិយាយថា គាត់ជាបុរសល្អ។ ពួកគេបាននិយាយថា «នេះគឺជារឿងជំនឿ ពីព្រោះថ្ងៃសប្បាតហ៍ ពាក់ព័ន្ធទៅនឹងជំនឿរបស់គាត់»។ ម្ចាស់ក្រុមហ៊ុនធំបានណែនាំ ដៃគូរបស់គាត់នៅប្រទេសឌូបៃ និងប្រទេស កេនយ៉ាឱ្យផ្អាកហោះហើរសិន។ គាត់បាននិយាយថា «វាជារឿងជំនឿ ដូច្នេះហើយ ទុកឱ្យគាត់កាន់តាម ជំនឿគាត់ចុះ»។

ក្នុងអំឡុងពេលនៃការជួលបួនខែ លោកខាសេរេខាបានហោះហើរជារៀងរាល់ថ្ងៃនៃសប្តាហ៍ លើកលែងតែថ្ងៃសប្បាតហ៍។ មួយឆ្នាំក្រោយមក ម្ចាស់ក្រុមហ៊ុនធំនៅឯអង់គ្លេស បានទូរស័ព្ទមកកាន់ លោកខាសេរេខាម្តងទៀត។ ម្ចាស់ក្រុមហ៊ុនបាននិយាយថា«ខ្ញុំចង់ជួបអ្នក មុខទល់នឹងមុខ ខ្ញុំចង់ទៅលេង ផ្ទះរបស់អ្នក ហើយមើលកន្លែងដែលអ្នករស់នៅ»។ ម្ចាស់ក្រុមហ៊ុនធំ បានធ្វើដំណើរទៅកាន់សាធារណរដ្ឋ ប្រជាធិបតេយ្យកុងហ្គោ ហើយស្នាក់នៅផ្ទះសំណាក់របស់លោកខាសេរេខា។ គាត់បាននិយាយថា «ខ្ញុំ មកជួបនឹងអ្នកនៅទីនេះ ដោយសារតែខ្ញុំមានរឿងចង់និយាយ។ ខ្ញុំចង់ដឹងថា តើខ្ញុំនឹងអាចរកបានអ្នក ដែល កាន់ជំនឿមុតមាំដូចជាអ្នក ដែលហ៊ានបដិសេធមិនព្រមហោះហើរនៅថ្ងៃសៅរ៍ដែរឬទេ»។ បន្ទាប់មក ម្ចាស់ ក្រុមហ៊ុនបានបង្ហាញអ្វីមួយ ដ៏គួរឱ្យភ្ញាក់ផ្អើល ដោយគាត់បាននិយាយថា «ខ្ញុំមិនដែលទទួលបានប្រាក់ចំណេញច្រើន ដូចដែលខ្ញុំបានធ្វើក្នុងអំឡុងពេលបួនខែនេះ ដែលក្រុមហ៊ុនយើងទាំង២បានធ្វើការជាមួយ គ្នានេះទេ។ ខ្ញុំចង់ធ្វើការជាមួយអ្នកម្តងទៀត ដោយមិនគិតប្រាក់ជួល»។ ម្ចាស់ក្រុមហ៊ុនធំបានផ្តល់យន្តហោះ

ឯកជន១៨កៅអី ដល់លោកខាសេរេខា។ លោកខាសេរេខាបានឆ្លើយតបទៅវិញ ទាំង បន្ទាបខ្លួន និងអស្ចារ្យនៅក្នុងចិត្តថា «ការខិតខំរបស់ខ្ញុំ គ្រាន់តែជាការបម្រើព្រះអម្ចាស់ របស់ខ្ញុំប៉ុណ្ណោះ»។
