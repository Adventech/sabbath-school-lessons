### ចូលលាក់ខ្លួននៅក្នុងសាកលវិទ្យាល័យសូលូស៊ី

នៅក្នុងប្រទេសសីមបាវេ លីនស៊ី (Lindsay) បានទៅរៀននៅឯសាកលវិទ្យាល័យសូលូស៊ី ដែលជាស្ថាប័នសេវេនដេយអាត់វេនទីស្ទ មិនមែនដោយសារតែជំនឿរបស់នាងនោះទេ។ នាងមិនមែន ជាអាត់វេនទីស្ទ ហើយនាងក៏មិនបានទៅរៀននៅឯសាកលវិទ្យាល័យនេះ ដោយសារតែនាងមានមិត្តភក្តិ នៅទីនោះដែរ។ នាងលីនស៊ីមិនស្គាល់នរណាម្នាក់ នៅក្នុងបរិវេណសាកលវិទ្យាល័យនេះឡើយ។ នាង លីនស៊ីបានចូលទៅរៀននៅក្នុងសាលានេះ ដោយសារនាងមិនចង់ឱ្យគេនិយាយមិនល្អអំពីនាង។ ខណៈដែលលីនស៊ីបានបញ្ចប់វិទ្យាល័យ នោះនាងមានបំណងចង់បន្តការសិក្សា នៅឯសាកល វិទ្យាល័យ។ ប៉ុន្តែ បន្ទាប់មក ឪពុកម្តាយរបស់នាងបានប្រាប់នាងថា ពួកគេមិនមានថវិកាឡើយ។ ម្តាយបាន និយាយថា «កូនត្រូវតែយល់អំពីចំណុចនេះ។ ម៉ាក់សន្យាថា កូននឹងបានទៅរៀននៅឯសាកលវិទ្យាល័យ បើទោះបីជាប៉ាមា៉ក់ ត្រូវចំណាយថវិកាអស់ប៉ុន្មានក៏ដោយ»។

ម្តាយដែលជាជាងកាត់ដេរ និងឪពុកដែលជាអ្នកលក់សម្លៀកបំពាក់ វាំងនន និងកម្រាលសាឡុង ពីក្រណាត់ដែលម្តាយបានដេរ។ ប៉ុន្តែ នៅពេលលីនស៊ីបញ្ចប់វិទ្យាល័យ អាជីវកម្មរបស់ពួកគេ មិនបាន ដំណើរការល្អនោះទេ ដូច្នេះហើយ ឪពុកម្តាយរបស់នាង បានសម្រេចចិត្តផ្លាស់ទៅនៅឯរដ្ឋបុតស្វាណាវិញ (Botswana)។ ឪពុកបានព្យាយាមលើកទឹកចិត្តនាង ដោយគាត់បាននិយាយថា «កុំបារម្ភអីកូនស្រី អ្វីៗ នឹងល្អប្រសើរឡើងវិញ»។

នាងលីនស៊ីបានជួយអាជីវកម្មនៅក្នុងគ្រួសារនាង អស់រយៈពេល៨ឆ្នាំបន្ថែមទៀត។ មិត្តភក្តិចាស់ៗបានទូរស័ព្ទពីប្រទេសសីមបាវេ មកសួរថា «តើឯងកំពុងធ្វើអ្វី?»។ មិត្តភក្តិទាំងនោះ បានសួរទៀត ថា «តើឯងកំពុងតែមានបញ្ហាមែនទេ?»។ នាងលីនស៊ីបានឆ្លើយទៅវិញថា «ឥឡូវនេះ ខ្ញុំកំពុងតែសិក្សា ដូចជាអ្នកទាំងអស់គ្នាដែរ»។

ចំណេរក្រោយមក នាងលីនស៊ីស្ទើរតែចង់បោះបង់ចោលក្តីស្រមៃ ដែលនាងចង់បន្តការសិក្សា នៅឯសាកលវិទ្យាល័យ។ ប៉ុន្តែ អាជីវកម្មនៃម្តាយរបស់នាង បានប្រសើរឡើងបន្តិចម្តងៗ។ នាងលីនស៊ី មានអតិថិជនកាន់តែច្រើនឡើង រហូតដល់នាងបានបើកហាងកាត់ដេរមួយកន្លែងទៀត។ នៅថ្ងៃមួយ ម្តាយ និងឪពុករបស់នាង បានហៅលីនស៊ីមកជួបនឹងពួកគេ ដោយប្រកាសទាំងញញឹមថា នាងអាចរើសយក សាកលវិទ្យាល័យណាមួយ ដើម្បីបន្តការសិក្សារបស់នាង។

នាងលីនស៊ីរំភើបចិត្តណាស់ ប៉ុន្តែ នាងមិនដឹងថា នាងត្រូវទៅរៀននៅឯសាកលវិទ្យាល័យមួយណានោះទេ។ នាងបានគិតថា នាងចង់ទៅរៀននៅឯសាកលវិទ្យាល័យសីមបាវេវិញ ប៉ុន្តែនាងមិនចង់ឱ្យមិត្តភក្តិដែលកំពុងតែសិក្សានៅទីនោះ ដឹងថានាងមិនបានចូលសាកលវិទ្យាល័យអស់រយៈពេល ជាច្រើនឆ្នាំ។ នាងលីនស៊ីបានរកមើលសាកលវិទ្យាល័យនៅក្នុងរដ្ឋមីដលែន (Midlands) ប៉ុន្តែ នាងក៏មាន មិត្តភក្តិសិក្សានៅទីនោះដែរ។ នាងមានមិត្តភក្តិនៅគ្រប់សាកលវិទ្យាល័យ លើកលែងតែសាកលវិទ្យាល័យ សូលូស៊ីប៉ុណ្ណោះ។ ដូច្នេះហើយ បានជានាងសម្រេចចិត្តទៅទីនោះ។

នៅពេលនាងទៅដល់ទីនោះ នាងលីនស៊ីមិនបានស្គាល់ព្រះវិហារអាត់វេនទីស្ទនោះឡើយ។ ប៉ុន្តែ នាងបានទៅព្រះវិហារជារៀងរាល់ថ្ងៃបរិសុទ្ធ បន្ទាប់មកទៀត នាងលីនស៊ីបានទទួលបុណ្យជ្រមុជទឹក។ ឪពុកម្តាយរបស់នាងបានអបអរសាទរចំពោះការសម្រេចចិត្តរស់នៅសម្រាប់ព្រះយេស៊ូវរបស់នាង។

សព្វថ្ងៃនេះ នាងលីនស៊ី ឈីកាន់ដា (Lindsay Chikanda) ដែលមានអាយុ២៤ឆ្នាំ កំពុងបញ្ចប់ ឆ្នាំទីមួយនៃការសិក្សារបស់នាង។ នាងបានត្រៀមខ្លួនជាស្រេច ដើម្បីប្រាប់មិត្តភក្តិវិទ្យាល័យរបស់នាងពីការពិត ដោយនិយាយថា «ឱ្យខ្ញុំសុំទោសផង ដែលកាលពីមុន ខ្ញុំមិនបានប្រាប់ការពិតដល់អ្នកទាំងអស់ គ្នា។ តាមពិតទៅ ខ្ញុំមិនបានសិក្សានៅឯសាកលវិទ្យាល័យណាមួយឡើយ។ ខ្ញុំបានផ្លាស់ទៅកាន់បុតស្វាណា ដោយសារតែឪពុកម្តាយរបស់ខ្ញុំ មិនមានលទ្ធភាពបង់ថ្លៃសិក្សាឱ្យខ្ញុំ ម្ល៉ោះហើយ បានជាខ្ញុំត្រូវជួយការងារពួកគាត់មួយរយៈ។ ប៉ុន្តែ បន្ទាប់មក ខ្ញុំបានមកចូលរៀននៅឯសាលកវិទ្យាល័យសូលូស៊ី ហើយខ្ញុំបានរកឃើញព្រះនៅទីនេះឯង។ តើអ្នកទាំងអស់គ្នាចង់ស្គាល់ព្រះ នៅក្នុងសាកលវិទ្យាល័យ នេះដែរឬទេ?»។
