title:  ការបៀតបៀនទាស់នឹងក្រុមជំនុំនៃគម្ពីរសញ្ញាថ្មី
date:  18/06/2025
---

គម្ពីរសញ្ញាថ្មីក៏បានជួយ ឱ្យយើងយល់ពីព្រឹត្តិការណ៍នៃគ្រាចុងក្រោយដែរ។ ជីវិតរបស់គី្រស្ទាន នៅក្នុងសម័យគម្ពីរសញ្ញាថ្មី មិនមែនជាជីវិតងាយស្រួលនោះទេ ដោយសារតែមានសាសន៍យូដាជាច្រើន បានស្អប់គ្រីស្ទាន ដោយសារតែពួកគេបានយល់ឃើញថា គ្រីស្ទានគឺជាគ្រោះ ដល់សាសន៍យូដារបស់ពួកគេ។ ទន្ទឹមគ្នានេះ នគររ៉ូមក៏បានធ្វើបាបគ្រីស្ទាន នៅក្នុងអំឡុងនៃសម័យនោះដែរ។ «អារក្ស និងមនុស្សលោក ដទៃទៀត បានរួមគ្នាបៀតបៀនសិស្សរបស់ព្រះយេស៊ូវ។ ដ្បិតនៅពេលណាដែលពួកគេបានធ្វើបាបសិស្ស របស់ទ្រង់ នោះមានន័យថា ពួកគេក៏បានធ្វើបាបទ្រង់ដែរ។ មនុស្សមួយចំនួនដែលបានបៀតបៀនសិស្ស របស់ព្រះយេស៊ូវ គឺជាសិស្សនៃព្រះក្លែងក្លាយ និងអ្នកដែលបានថ្វាយបង្គំធម្មជាតិនោះឯង។ មនុស្សទាំងនេះ បានយល់ឃើញថា បើសិនជាអ្នកណាម្នាក់បានទទួលយក ដំណឹងល្អអំពីព្រះយេស៊ូវ នោះមានន័យថា ពួកគេនឹងឈប់ទៅថ្វាយបង្គំព្រះក្លែងក្លាយនៅក្នុងវិហាររបស់ពួកគេ ហើយពួកគេក៏ឈប់ថ្វាយតង្វាយដល់ ព្រះក្លែងក្លាយ នៅលើអាសនារបស់ពួកគេនោះដែរ។ ដូច្នេះហើយ សិស្សរបស់ព្រះក្លែងក្លាយទាំងនេះ បាន សម្រេចចិត្តថា ពួកគេនឹងធ្វើឱ្យសាសនាគ្រិស្តវិនាសបង់»។ ដកស្រង់ពី Ellen G. White, The Great Controversy, page 39, adapted។

រឿងនៅក្នុងគម្ពីរកិច្ចការ (កិច្ចការ 12:1-17) បានបង្ហាញយើង នូវអ្វីដែលរាស្ត្ររបស់ព្រះអាច នឹងរំពឹងថា នឹងកើតមានឡើងនៅក្នុងសម័យចុងក្រោយនោះ ទៅតាមវិធីមួយដ៏មានឥទ្ធិពល (សូមអាន បទគម្ពីរវិវរណៈ ជំពូក13)។

`សូមអានគម្ពីរកិច្ចការ 12:1-17។ តើនៅក្នុងសាច់រឿងនេះ មានផ្នែកណាខ្លះ ដែលបានជួយ ឱ្យលោកអ្នកយល់ពីរឿងមួយចំនួន ដែលត្រូវកើតឡើងនៅក្នុងសម័យចុងក្រោយនោះ?`

ស្តេចហេរ៉ូឌអាគ្រីប៉ាបានកាត់ក្បាលរបស់លោកយ៉ាកុប។ លោកពេត្រុសគឺជាអ្នកដែលបានស្លាប់បន្ទាប់ពីលោកយ៉ាកុប។ នៅក្នុងសម័យនោះ លោកអ្នកនឹងត្រូវគេសម្លាប់ ដោយសារតែលោកអ្នក គឺជាអ្នកគ្រីស្ទាន។ ផ្នែកដ៏ភ្ញាក់ផ្អើលបំផុតនៅក្នុងសាច់រឿងនេះ គឺថា លោកពេត្រុសកំពុងតែដេកលង់លក់ នៅក្នុងអំឡុងនៃយប់ចុងក្រោយ នៃឆាកជីវិតរបស់គាត់។ គាត់បានគេងយ៉ាងលង់លក់ រហូតដល់ «ទេវតា បំបាក់ឆ្អឹងជំនីររបស់គាត់» ដើម្បីឱ្យគាត់បានភ្ញាក់ពីដំណេក។

ជាការពិតណាស់ ទេវតាបានដោះលែងលោកពេត្រុស ឱ្យរួចផុតពីការឃុំឃាំង។ បន្ទាប់មក លោកពេត្រុសបានទៅជួបនឹងគ្រីស្ទាន ដែលកំពុងតែអធិស្ឋានឱ្យគាត់។ ប៉ុន្តែ នៅពេលដែលលោកពេត្រុស បានបង្ហាញខ្លួនគាត់ឱ្យពួកគេបានឃើញ នោះគ្រីស្ទានទាំងនេះបែរជាពិបាកនឹងជឿថា លោកពេត្រុសបាន មានសេរីភាពទៅវិញ។ ព្រះគម្ពីរបានចែងថា ពួកគេបានភ្ញាក់ផ្អើលជាខ្លាំង។ តើយើងធ្លាប់ដែលអធិស្ឋាន ទូលសូមរបស់អ្វីមួយ ដូចជាអ្នកគ្រីស្ទានទាំងនេះ ប៉ុន្តែ ពិបាកនឹងជឿថា ព្រះនឹងឆ្លើយតបនឹងយើងនោះញឹកញាប់ប៉ុនណាដែរ?

នៅក្នុងអំឡុងនៃសម័យនោះ មានគ្រីស្ទានខ្លះ ត្រូវបានគេរំដោះឱ្យរួចផុតពីការស្លាប់។ ប៉ុន្តែ ក៏មានគ្រីស្ទានខ្លះ ត្រូវគេសម្លាប់ដែរ។ ទន្ទឹមគ្នានេះ ខណៈពេលដែលយើងខិតជិតមកដល់ចុងបញ្ចប់នៃប្រវត្តិសាស្ត្រមនុស្សលោកនេះ នោះរឿងដដែលនេះក៏នឹងបានកើតឡើងដែរ។ ដូច្នេះហើយ ក្រោយមក ទៀត លោកពេត្រុសក៏បានស្លាប់ ដោយសារតែជំនឿដែលគាត់មាននោះដែរ (សូមអានគម្ពីរយ៉ូហាន 21: 18, 19)។
