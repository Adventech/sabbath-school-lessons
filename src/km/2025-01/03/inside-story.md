### អ្នកជំនួយការដែលបានបង្ហាញពីការដឹងគុណ

លោកចុងមូន យូន (Jong-Moon Yoon) ត្រូវបានគេស្គាល់ថា ជាអ្នកជំនួយការដែលបាន ដឹងគុណ។ មិនថាកាលៈទេសៈបែបណានោះទេ ពាក្យពេចន៍ដំបូងគេដែលចេញពីបបូរមាត់របស់គាត់ នោះ គឺជាពាក្យថា «អរគុណ»។ នៅពេលដែលអ្នកណាម្នាក់មកសុំជំនួយពីគាត់ នោះគាត់បាននិយាយថា «អរគុណ»។ នៅពេលដែលគាត់ប្រឈមនឹងជម្លោះ នោះគាត់បាននិយាយថា «អរគុណ»។ ជាពិសេស ទៅទៀតនោះ នៅពេលដែលគាត់មានការលំបាក និងការខាតបង់ នោះគាត់ក៏បាននិយាយថា «អរគុណ» ដែរ។

អ្នកជំនួយការនៅក្នុងព្រះវិហារសេវេនដេយអាត់វេនទីស្ទនៅឯប្រទេសកូរ៉េខាងត្បូង បានផ្តក់ចិត្តរបស់គាត់ទៅនឹងពាក្យពេចន៍របស់លោកប៉ុលនៅក្នុងគម្ពីរថែស្សាឡូនីចទី១ ៥:១៨ ដែលបានមាន ប្រសាសន៍យ៉ាងដូច្នេះថា «ចូរអរព្រះគុណក្នុងគ្រប់កាលៈទេសៈទាំងអស់ ដ្បិតព្រះសព្វព្រះហឫទ័យឲ្យ អ្នករាល់គ្នាធ្វើដូច្នេះក្នុងព្រះគ្រីស្ទយេស៊ូវ»។ ដូច្នេះហើយ មនុស្សគ្រប់គ្នាបានកត់សម្កាល់ពីការដឹងគុណរបស់គាត់។ លោកចុងមូន គឺជាម្ចាស់ហាងបាយចំណាស់មួយនៅលើឆ្នេរមួយកន្លែងនៅក្នុងទីក្រុងជង់ដួងជីង (Jeongdongjin) ដែលជាកន្លែងមើលព្រះអាទិត្យរះយ៉ាងល្បីល្បាញនៅក្នុងប្រទេសកូរ៉េខាងត្បូង ជាពិសេសទៅទៀតនោះ គឺនៅក្នុងថ្ងៃចូលឆ្នាំថ្មី។ ស្រី្តណាមនំ (Nanum) ជាអតិថិជនម្នាក់របស់គាត់នៅក្នុង ចំណោមអតិថិជនទាំងអស់។ នៅក្នុងយុវភាពរបស់នាង នាងបានចំណាយពេលម្តងម្កាលងើយមើល ព្រះអាទិត្យរះដោយក្តីអំណរ ជាមួយទឹកតែ សុាំងវិចស្នូលពងទា សាឡាត់ និងឈីស (ប្រហុកបារាំង)។ នៅពេលដែលនាងធំឡើង នាងបានរៀបអាពាណ៍ពិពាហ៍ និងមានកូនស្រីម្នាក់ឈ្មោះថា អាស្សា (Azsa)។ នាង និងកូនស្រីនាងបានក្លាយមកជាអតិថិជនរបស់លោកចុងមូនដែលបានបរិភោគអាហារពេលព្រឹក និងបានចាប់អារម្មណ៍ទៅលើពាក្យអរគុណរបស់លោកចុងមូន។ នៅពេលដែលនាងកុម៉្មង់អាហារ នោះ លោកចុងមូនបាននិយាយថា «អរគុណ»។ នៅពេលដែលនាងបង់ថ្លៃអាហារ នោះលោកចុងមូនបាន និយាយថា «អរគុណ»។ នៅពេលនាងនិយាយថា អរគុណសម្រាប់អាហារ នោះលោកចុងមូនក៏បាន និយាយថា អរគុណដែរ។

ថ្ងៃមួយ នៅពេលដែលនាងទៅលេងហាងបាយរបស់គាត់ នោះនាងបានសំណេះសំណាល ជាមួយនឹងគាត់ និងបានដឹងថា គាត់បានទៅព្រះវិហារជារៀងរាល់ថ្ងៃសោរ៍។ នាងមិនដែលបានឮអំពី អ្នកថ្វាយបង្គំនៅក្នុងថ្ងៃសៅរ៍ពីមុនមកនោះទេ។ នាងបានសួរថា «តើខ្ញុំអាចទៅព្រះវិហារជាមួយនឹងបងប្រុស បានដែរឬទេ?»។ នៅថ្ងៃសៅរ៍បន្ទាប់ នាង និងកូនស្រីអាយុ៥ឆ្នាំរបស់នាង បានរួមដំណើរជាមួយនឹងលោក ចុងមូនទៅព្រះវិហាររបស់គាត់ ដែលត្រូវចំណាយពេលបើកបរប្រហែលជា៣០នាទីពីហាងបាយរបស់គាត់។ សមាជិកក្រុមជំនុំបានស្វាគមន៍នាង និងកូនស្រីនាងយ៉ាងរាក់ទាក់ ដោយពួកគេបានចង់ដឹងពីដំណើរ ដើមទងដែលនាំឱ្យនាង និងកូនស្រីនាងស្គាល់ព្រះវិហារនោះ។ នាងបាននិយាយថា ខ្ញុំបានស្គាល់កន្លែងនេះ ដោយសារតែអ្នកជំនួយការព្រះវិហារដែលបានបង្ហាញពីការដឹងគុណរបស់គាត់នោះឯង។

បន្ទាប់ពីកម្មវិធីថ្វាយបង្គំព្រះរួចមក នោះនាងណាមនំបានប្រាប់លោកចុងមូនថា នាងចង់មក ថ្វាយបង្គំម្តងទៀត។ នាងបាននិយាយថា «បើសិនជាព្រះវិហារនេះជាព្រះវិហារដែលបងប្រុសតែងតែមក ថ្វាយបង្គំ នោះខ្ញុំនឹងមកថ្វាយបង្គំតាមបងប្រុសដែរ»។ ទោះបីជានាងណាមនំមិនដែលបានឮព្រះនាមរបស់ព្រះយេស៊ូវនៅក្នុងហាងបាយរបស់លោកចុងមូនក៏ដោយ ក៏នាងបានឃើញព្រះយេស៊ូវតាមរយៈវិញ្ញាណនៃការដឹងគុណរបស់លោកចុងមូនដែរ។ ដំណឹងល្អមិនមែនត្រឹមតែជាពាក្យសម្តីនោះទេ ប៉ុន្តែ ដំណឹងល្អក៏បានបង្ហាញតាមរយៈការរស់នៅរបស់បុគ្គលម្នាក់ៗផងដែរ។ ដូច្នេះហើយ តាមរយៈគំរូរបស់

លោកចុងមូននេះ ម្តាយវ័យក្មេង និងកូនស្រីរបស់គាត់បានចាប់ផ្តើមមកព្រះវិហារជា ប្រចាំ។ សព្វថ្ងៃនេះ នាងកំពុងតែអធិស្ឋាន សូមឱ្យស្វាមីរបស់នាងបានចូលរួមថ្វាយបង្គំជារៀងរាល់ព្រឹកថ្ងៃសប្បាតហ៍ជាមួយ នឹងពួកគេផងដែរ។
