### ត្រូវបានធ្វើបាបដោយអ្នកស្រវឹងស្រា

បុរសស្រវឹងម្នាក់បានចាប់បេសកនារីអាយុ២១ឆ្នាំនៅលើរថយន្តក្រុង ដោយមានបំណងចង់ ព្យាយាមឱបថើបនាង។ នៅពេលដែលនាងព្យាយាមប្រកែកទាស់នឹងការប៉ុនប៉ងរបស់គាត់ នោះអ្នកដំណើរ ដទៃទៀតមិនបានចាប់អារម្មណ៍ឡើយ រហូតដល់នាងស្រែកប្រាប់អ្នកបើករថយន្តក្រុង។ រថយន្តក្រុងបាន ឈប់ បន្ទាប់មក អ្នកដំណើរបានដេញបុរសម្នាក់នោះចេញពីរថយន្ត ដោយមានស្ត្រីចំណាស់ម្នាក់បានឃើញ ហេតុការណ៍នោះ។ ស្ត្រីចំណាស់ម្នាក់នោះបានចូលមកលួងលោមបេសកនារី ដោយកាន់ដៃនាង រួចនិយាយ ថា នូវពាក្យពេចន៍មួយចំនួនដែលបេសកនារីមិនអាចយល់បាន។

នៅក្នុងប្រទេសម៉ុងហ្គោលី ហេតុការណ៍បែបនេះគឺជាសុបិនអាក្រក់ដែលនាងចូអាន ប៉ាក គីម (Joane Park Kim) បានប្រឈមសឹងតែរាល់សប្តាហ៍។ នៅដើមឆ្នាំ១៩៩០ នាងចូអានគឺជាស្ត្រីជនជាតិ អាមេរិកកាំងតែម្នាក់ប៉ុណ្ណោះ ដែលបានបម្រើការជាបេសកនារីសេវេនដេយអាត់វេនទីស្ទដំបូងគេបង្អស់នៅក្នុងប្រទេសម៉ុងហ្គោលីដែលមានភាពខ្វះខាតក្រោយពីសម័យកុម្មុយនិស្ត។

មានម្តងនោះ អ្នកជិតខាងដែលបានស្រវឹងស្រាបានចូលទៅក្នុងផ្ទះរបស់នាង ដោយនឹកស្មាន ថាជាផ្ទះរបស់គាត់។ គាត់បានធាក់ទ្វារឈើដ៏ស្តើងរបស់នាងដោយប្រើស្បែកជើងកែង រួចបានចូលទៅ វាយដំនាង។ នាងចូអានបានស្រែក រួចទាញអំបោសមកវាយតវិញ។ នាងចូអានមិនអាចយកឈ្នះបុរស ស្រវឹងស្រាម្នាក់នោះបានទេ នោះនាងបានយកអំបោសវាយដំបូលផ្ទះខ្លាំងៗ បន្ទាប់មក ស្វាមី និងភរិយា ដែលជាបេសកជនមួយគូទៀតនៅផ្ទះខាងលើនាង បានប្រញាប់ប្រញាល់រត់មកជួយនាង។

ឧបសគ្គចុងក្រោយសម្រាប់លោកចូអានបានមកដល់ នៅពេលដែលនាង និងបេសកនារីម្នាក់ ទៀតកំពុងតែរង់ចាំរថយន្ត នៅត្រង់ចំណតរថយន្តក្រុងនៃជាយក្រុងអ៊ូឡាន់បាតារ ដែលជារាជធានីនៃប្រទេសម៉ុងហ្គោលី។ ម្តាយនៃក្មេងម្នាក់បានអញ្ជើញពួកគេ ឱ្យចូលរួមនៅក្នុងពិធីកាត់សក់កូននាងជាលើក ដំបូង ដែលការកាត់សក់នេះ គឺជាពិធីធំនៅក្នុងប្រទេសម៉ុងហ្គោលី។ ម្តាយនៃក្មេងនោះបានប្រាប់ឱ្យនាង រង់ចាំនៅត្រង់ចំណតរថយន្តក្រុង ដោយសារផ្ទះរបស់ស្រីម្នាក់នោះពុំមានដៅនៅក្នុងប្រព័ន្ធផែនទី ដូច្នេះ ហើយ បានជានាង និងបេសកនារីម្នាក់ទៀតបានអង្គុយនៅលើបង់រង់ចាំស្រ្តីម្នាក់នោះមកយកពួកគេវិញ។

បន្ទាប់មក បុរសស្រវឹងស្រាទាំង៤នាក់បានអង្គុយនៅក្បែរពួកគេ រួចបានព្យាយាមឱបថើបនាង ចូអាន។ នាងចូអាន និងបេសកនារីម្នាក់ទៀតបានខិតចេញពីពួកគេ ប៉ុន្តែ បុរសស្រវឹងទាំងនោះបានខិត ជិតពួកគេវិញ។ ហ្វូងមនុស្សដែលនៅត្រង់ចំណងរថយន្តក្រុងមិនបានចាប់អារម្មណ៍ឡើយ។ ពួកបុរសទាំងនោះបានចាប់ទាញចូលទៅក្នុងផ្លូវតូច។ នៅពេលនោះ នាងចូអានបានស្រែក និងព្យាយាមរើបម្រាស់ ប៉ុន្តែ នាងមិនអាចទប់ទល់នឹងបុរស៤នាក់នោះបានទេ។ បន្ទាប់មក បុរសទាំងនោះបានច្រាននាងទៅលើដី ធ្វើឱ្យនាងគិតនៅក្នុងចិត្តថា ពួកគេនឹងចាប់រំលោភនាង។

ប៉ុន្តែ រំពេចនោះ មុខរបស់បុរសទាំងនោះបានប្រែជាស្លេកស្លាំង។ ពួកគេថយចេញ រួចរត់អស់ ទៅ។ នាងចូអានបានក្រឡេកមើលជំុវិញនាង ដើម្បីរកមើលអ្នកដែលបានមកជួយពួកគេ។ ប៉ុន្តែ ពួកគេមិនបានឃើញអ្នកណាម្នាក់នោះឡើយ។ នៅពេលនោះ នាងចូអានបានដឹងភ្លាមថា បុរសទាំងនោះប្រាកដ ជាបានឃើញទេវតាហើយ។
