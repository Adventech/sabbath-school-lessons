### ព្រះនៃសេចក្តី ស្រឡាញ់ និង យុត្តិធម៌

ព្រះជាសេចក្តីស្រឡាញ់។ ចំណុចនេះជាអ្វីដែលយើងបានអាននៅក្នុងគម្ពីរយ៉ូហានទី១ ៤:៨ និងជំពូក ១៦។ ព្រះគម្ពីរទាំងមូលបានប្រាប់យើងថា ព្រះយេស៊ូវជាសេចក្តីស្រឡាញ់។ ហើយសេចក្តីស្រឡាញ់របស់ ទ្រង់ គឺជាដើមចមនៃសេចក្តីជំនឿគ្រីស្ទាន។ សេចក្តីស្រឡាញ់ជាអត្តសញ្ញាណរបស់ព្រះ។ សេចក្តីស្រឡាញ់ ជាផ្នែកនៃអ្វីៗគ្រប់យ៉ាងដែលយើងបានជឿ និងបានធ្វើ។ ប្រសិនបើយើងបានយល់ឃើញថា យើងចាំបាច់ ត្រូវទិញសេចក្តីស្រឡាញ់របស់ព្រះ នោះយើងអាចនឹងជឿថា ព្រះមិនបានស្រឡាញ់យើងឡើយ ដោយសារ តែយើងជាមនុស្សមិនស័ក្តិសម និងជាអ្នកដែលមានពេញទៅដោយអំពើបាប។ ការយល់ឃើញបែបនេះ គឺជាកំហុសឆ្គងមួយ។

អ្វីដែលយើងបានជឿ និងរបៀបដែលយើងបានរស់នៅតាមជំនឿរបស់យើង គឺពឹងផ្នែកទៅលើការយល់ដឹងពីសេចក្តីស្រឡាញ់របស់ព្រះ។ ដូច្នេះហើយ បានជាយើងត្រូវសួរសំណួរថា៖ តើអ្វីទៅជាសេចក្តី ស្រឡាញ់? ប្រសិនបើលោកអ្នកសួរមនុស្ស១០នាក់ឱ្យឆ្លើយសំណួរនេះ នោះលោកអ្នកនឹងទទួលបានចម្លើយ១០យ៉ាងផ្សេងៗពីគ្នា។ ប៉ុន្តែ ក៏គួរឱ្យសោកស្តាយណាស់ដែរ ដែលមានគ្រីស្ទានជាច្រើននាក់បាន មានគោលគំនិត និងគោលជំនឿខុសឆ្គងអំពីសេចក្តីស្រឡាញ់នេះ។

អ្នកគ្រីស្ទានបានផ្តល់ចម្លើយខុសៗគ្នា ដើម្បីឆ្លើយតបទៅនឹងសំណួរទាំងនេះ៖

«តើសេចក្តីស្រឡាញ់របស់ព្រះមានតែការឱ្យមួយមុខ តែមិនមានការទទួលនោះ មែនទេ?»

«តើសេចក្តីស្រឡាញ់របស់ព្រះ ត្រឹមតែស្តីអំពីការសុគតរបស់ទ្រង់នៅលើឈើឆ្កាង មែនទេ?»

«តើព្រះបានសព្វព្រះទ័យជាមួយនឹងមនុស្សលោកដែរឬទេ?»

តើព្រះពិតជាបានយកចិត្តទុកដាក់អំពីមនុស្សដែរឬទេ? តើមនុស្សអាចបដិសេធសេចក្តីស្រឡាញ់ របស់ព្រះ ឬលះបង់សេចក្តីស្រឡាញ់នេះបានដែរឬទេ?»

តើព្រះអាចខ្ញាល់ដែរឬទេ បើសិនជាទ្រង់ជាសេចក្តីស្រឡាញ់នោះ?»

តើសេចក្តីស្រឡាញ់ និងភាពយុត្តិធម៌ បានផ្សាភ្ជាប់គ្នាដោយរបៀបណា?»

«តើមនុស្សលោកអាចស្រឡាញ់អ្នកដទៃដូចជាព្រះបានដែរឬទេ?» «បើដូច្នេះមែន តើសេចក្តី ស្រឡាញ់នឹងមានសភាពយ៉ាងណាដែរ?»

ចម្លើយឆ្លើយតបទៅនឹងសំណួរទាំងនេះ មើលទៅហាក់ដូចជាងាយស្រួល។ ប៉ុន្តែ ចម្លើយខ្លះមិនបាន ស្របទៅនឹងអ្វីដែលព្រះគម្ពីរបានបង្រៀន អំពីសេចក្តីស្រឡាញ់របស់ព្រះនោះទេ។

យើងនឹងឆ្លើយសំណួរទាំងអស់នេះ នៅក្នុងពេលតែមួយ។ ពោលគឺយើងនឹងក្រឡេកមើលសំណួរទាំងនេះ និងសំណួរដទៃទៀត នៅក្នុងអំឡុងពេលដែលយើងសិក្សាមេរៀននៅក្នុងត្រីមាសនេះ។ យើង នឹងបានឃើញថា សេចក្តីស្រឡាញ់របស់ព្រះ សំខាន់ជាងការដែលយើងបានយល់ឃើញទៅទៀត។ ដ្បិត សេចក្តីស្រឡាញ់របស់ព្រះដែលមានចែងនៅក្នុងព្រះគម្ពីរ នោះមានអំណាច និងស៊ីជម្រៅជាងគោលគំនិត ពេញនិយម អំពីសេចក្តីស្រឡាញ់នៅពេលបច្ចុប្បន្ននេះទៅទៀត។ ដូច្នេះហើយ ពីមួយសប្តាហ៍ទៅមួយ សប្តាហ៍ យើងនឹងក្រឡេកមើលយ៉ាងប្រុងប្រយ័ត្ន អំពីផ្នែកសំខាន់ៗ និងភាពស្រស់ស្អាតមួយចំនួនពី ស្រឡាញ់របស់ព្រះដែលមានចែងនៅក្នុងព្រះគម្ពីរ។

ខណៈដែលយើងបន្តនូវការសិក្សារបស់យើង នោះយើងនឹងបានឃើញពីរបៀបដែលព្រះជាសេចក្តីស្រឡាញ់ផង និងជាសេចក្តីយុត្តិធម៌ផង។ ព្រះគម្ពីរបង្ហាញថា រឿងទាំង២យ៉ាងអំពីព្រះនេះ មិនអាចបំបែក ពីគ្នាបានឡើយ។ ព្រះនៃព្រះគម្ពីរ គឺជាព្រះនៃសេចក្តីយុត្តិធម៌ ពីព្រោះទ្រង់ស្រឡាញ់នូវការសម្រេចចិត្តដែលយុត្តិធម៌ (អេសាយ ៦១:៨)។ ដោយសារតែព្រះជាសេចក្តីស្រឡាញ់នេះហើយ បានជាទ្រង់យក ព្រះទ័យទុកដាក់ចំពោះការឈឺចាប់នៅលើផែនដីនេះ។ ទ្រង់មានព្រះទ័យអាណិតអាសូរចំពោះអ្នកដែល គេបានធ្វើបាប និងអ្នកដែលស្ថិតនៅក្នុងការឈឺចាប់។ ព្រះទ្រង់ជ្រាបពីការឈឺចាប់របស់ពួកគេ នៅក្នុង ព្រះទ័យរបស់ទ្រង់។ សេចក្តីអាក្រក់បានបង្កការឈឺចាប់នេះដល់មនុស្សលោក និងដល់ព្រះ។ ប៉ុន្តែ ព្រះ ទ្រង់ឈឺចាប់ ជាងមនុស្សអាក្រក់ដទៃទៀត ដែលអំពើបាបបានបង្កឡើងទៅទៀត។

ព្រះគម្ពីរបានបង្ហាញយើងម្តងហើយម្តងទៀតថា ព្រះទ្រង់ស្រឡាញ់យើងយ៉ាងជ្រាលជ្រៅ នៅពេល ដែលព្រះយេស៊ូវមានបន្ទូល «ឱយេរូសាឡិម យេរូសាឡិមជាក្រុងដែលសម្លាប់ពួកហោរា ហើយយកថ្មគប់ពួកអ្នកដែលព្រះបានចាត់ឲ្យមករកអ្នកអើយ! តើប៉ុន្មានដងហើយដែលខ្ញុំចង់ប្រមូលកូនរបស់អ្នកមក ដូចមេមាន់ប្រមូលកូនរបស់វាមកជ្រកក្រោមស្លាប តែអ្នកមិនព្រមទេ» (ម៉ាថាយ ២៣:៣៧)។

ព្រះនៃព្រះគម្ពីរ គឺជាព្រះនៃសេចក្តីស្រឡាញ់។ ព្រះគម្ពីរបានបង្ហាញយើងថា ព្រះបានអន់ព្រះទ័យ ជាញឹកញាប់។ ទ្រង់មានព្រះទ័យសោកសៅ។ ទ្រង់មានអារម្មណ៍ថា មនុស្សលោកបានបដិសេធទ្រង់។ រឿងទាំងអស់ដែលមានចែងនៅក្នុងព្រះគម្ពីរ គឺស្តីអំពីកិច្ចការគ្រប់យ៉ាងដែលព្រះបានធ្វើ និងកំពុងតែធ្វើ នៅពេលឥឡូវនេះ ដើម្បីបំពេញសេចក្តីស្រឡាញ់នៅគ្រប់ទាំងផ្នែក និងគ្រប់ទីកន្លែងនៅលើផែនដីនេះ។ រឿងនេះ និងរឿងជាច្រើនទៀត ជាប្រធានបទសម្រាប់ការសិក្សាព្រះគម្ពីរនៅក្នុងត្រីមាសនេះ។
