### បានជាប់គាំងនៅក្នុងតំបន់មជ្ឈិមបូព៌ា

ជីវិតរបស់នាងគីណាម (Ki-nam) បានធ្លាក់ចុះយ៉ាងខ្លាំង បន្ទាប់ពីនាងបានមកដល់ប្រទេសកូរ៉េ ខាងត្បូង ដើម្បីបម្រើការជាបេសកជននិស្សិតនៅក្នុងតំបន់មជ្ឈិមបូព៌ា។ នាងបានមកដល់ប្រទេសនេះដោយ ប្រើទិដ្ឋាការ (Visa) ស្នាក់នៅមួយរយៈពេលខ្លី ប៉ុន្តែ ទិដ្ឋាការនោះចាំបាច់ត្រូវតែរត់ការឱ្យក្លាយទៅជាទិដ្ឋាការ ស្នាក់នៅរយៈពេលវែងវិញ បាននាងអាចស្នាក់បាន១ឆ្នាំពេញ។ ម្ចាស់នៃផ្ទះជួលរបស់នាងបានយល់ព្រម ចុះហត្ថលេខាលើឯកសារ ដើម្បីបញ្ជាក់ពីការស្នាក់នៅសម្រាប់ទិដ្ឋាការថ្មីរបស់នាង។ ប៉ុន្តែ រំពេចននោះ ម្ចាស់ផ្ទះបានប្តូរចិត្តរបស់គាត់ ដោយទាមទាររកថ្លៃទឹកចិត្តយ៉ាងច្រើន បានជាគាត់យល់ព្រមចុះហត្ថលេខា លើឯកសារនោះ។

នាងគីណាមពុំមានប្រាក់ជាប់ខ្លួននោះទេ ពីព្រោះអ្នកដឹកនាំក្រុមបេសកជនដែលបានមើលការខុសត្រូវចំពោះក្រុមបេសកជននិស្សិត ក៏មិនមានប្រាប់ដែរ។ លើសពីនេះទៀតនោះ សហគមន៍សេវេនដេយ អាត់វេនទីស្ទដ៏តូចរបស់ពួកគេ ក៏គ្មានប្រាប់នោះដែរ ពីព្រោះសហគមន៍នេះមានសុទ្ធតែជនភៀសខ្លួនខ្សត់ ខ្សោយ ដែលបានថ្វាយបង្គំនៅក្នុងព្រះវិហារនៅតាមផ្ទះរបស់ពួកគេ។ ជម្រើសរបស់នាងគីណាមមានដែន កំណត់។ នាងមិនអាចរើទៅនៅឯផ្ទះជួលមួយទៀតបានទេ ដោយសារតែវាពិបាកនឹងស្វែងរកម្ចាស់ផ្ទះដែល ចង់ជួលផ្ទះឱ្យជនបរទេសស្នាក់នៅនោះណាស់។ ហើយការរស់នៅនៅក្នុងប្រទេសកូរ៉េខាងត្បូងដោយគ្មាន ច្បាប់នោះ ក៏មិនមែនជាជម្រើសល្អដែរ ហើយលើសពីនេះទៅទៀតនោះ នាងក៏មិនចង់សូកប៉ាន់មន្ត្រីដើម្បី ឱ្យនាងទទួលបានទិដ្ឋាការស្នាក់នៅនោះដែរ។

នាងបានអធិស្ឋានថា «ឱព្រះអម្ចាស់អើយ បើសិនជាទ្រង់បានចាត់ទូលបង្គំឱ្យមកប្រទេសនេះ វិញ នោះទ្រង់ក៏គួរតែមានដំណោះស្រាយចំពោះបញ្ហារបស់ទូលបង្គំដែរ»។ នាងបានអធិស្ឋានជារៀងរាល់ថ្ងៃ អស់រយៈពេល២ខែ។ ហើយឪពុកម្តាយរបស់នាងដែលស្នាក់នៅក្នុងប្រទេសកូរ៉េខាងត្បូង ក៏បានអធិស្ឋាន រឿងនេះដែរ។ អ្នកដឹកនាំក្រុមបេសកជនបានដាក់ឈ្មោះនាងគីណាម ទៅក្នុងបញ្ជីអធិស្ឋាននៅក្នុងព្រះវិហារ នៅតាមផ្ទះ បន្ទាប់មក សមាជិកក្រុមជំនុំបាននាំគ្នាអធិស្ឋានឱ្យនាង។

២ខែក្រោយមក ម្ចាស់ផ្ទះជួលមិនបានចុះហត្ថលេខាលើឯកសាររបស់នាងនោះទេ។ បន្ទាប់មក ថ្ងៃដែលនាងគីណាមត្រូវទៅធ្វើកិច្ចសម្ភាសន៍ស្នើសុំទិដ្ឋាការថ្មីនោះក៏បានមកដល់។ នៅពេលនោះ ទូរស័ព្ទ របស់នាងគីណាមបានរោទិ៍ឡើង មុនពេលដែលនាងបម្រុងនឹងចូលទៅសម្ភាសន៍។ អ្នកដែលទូរស័ព្ទមក នាង គឺជាអ្នកដឹកនាំនៃក្រុមបេសកជន។ គាត់បាននិយាយថា «មីណាមចេញពីកន្លែងទិដ្ឋាការសិនមក ពួកយើងនៅមានដំណោះស្រាយមួយទៀត»។ គាត់បានបកស្រាយថា មុននេះបន្តិច មានសមាជិកក្រុមជំនុំ ម្នាក់បានចូលទៅជួបនឹងម្ចាស់អចលនទ្រព្យ សួរនាំអំពីផ្ទះជួលរបស់នាងហើយ។ ក្រុមជំនុំម្នាក់នេះបាន ស្គាល់ម្ចាស់អចលនទ្រព្យ បន្ទាប់មក គាត់បានសួរថា «តើអ្នកអាចជួយមិត្តភក្តិខ្ញុំបានដែរឬទេ?»។ ម្ចាស់ផ្ទះជួលរបស់នាងបានតបមកវិញថា «ឱ្យនាងយកលិខិតឆ្លងដែនមកឱ្យខ្ញុំមក ចាំខ្ញុំចុះហត្ថលេខាបញ្ជាក់ពី ការស្នាក់នៅរបស់នាង»។ នាងគីណាមបានភ្ញាក់ផ្អើលជាខ្លាំង។ នាងអាចនិយាយបានត្រឹមតែពាក្យថា «អរគុណព្រះអង្គ» ប៉ុណ្ណោះ។ ម្ចាស់ផ្ទះបានចុះហត្ថលេខានៅលើឯកសាររបស់នាង បន្ទាប់មក នាងក៏បាន ទទួលទិដ្ឋាការស្នាក់នៅ។ បន្ទាប់ពីនោះមក នាងគីណាមក៏គ្មានអ្វីទៀតដែរដែលថា ព្រះនឹងប្រទានពរដល់ នាងនៅក្នុងរយៈពេលមួយឆ្នាំដែលនាងបានស្នាក់នៅឯតំបន់មជ្ឈិមបូព៌ានោះ។ ហើយទ្រង់ក៏បានប្រទាន ពរដល់នាងមែន ដោយមានមនុស្សចំនួន៧នាក់បានជ្រមុជទឹកតាមរយៈកិច្ចការរបស់នាង។ នាងគីណាម បាននិយាយនៅក្នុងកិច្ចសម្ភាសន៍ ជាមួយនឹងមិសិនអាត់វេនទីស្ទនៅក្នុងទីក្រុងសេអ៊ូលនៃប្រទេសកូរ៉េខាង ត្បូងយ៉ាងដូច្នេះថា «ព្រះបានត្រាស់ហៅខ្ញុំ និងបានប្រើខ្ញុំឱ្យទៅសង្គ្រោះអ្នកដទៃ ទ្រង់បានគង់នៅជាមួយ នឹងខ្ញុំគ្រប់ជំហានទាំងអស់ ដើម្បីជួយដល់ខ្ញុំ។
