### «យើងមិនបានបញ្ជូនអ្នកណាម្នាក់នោះឡើយ»

លោកដូណាល់ដូ វេឡាស្វីស (Donaldo Velasquez) បានសួរសុខទុក្ខអ្នកទោសជនជាតិកូឡុំប៊ី ជារៀងរាល់ថ្ងៃសុក្រ ដើម្បីចែកចាយអំពីព្រះយេស៊ូវ។ ប៉ុន្តែ នៅថ្ងៃសុក្រ គាត់មិនបានចូលទៅកន្លែងនោះ បានឡើយ ពីព្រោះមានគេត្រូវការឱ្យគាត់ជួយ ដោយសារតែគាត់បានធ្វើការជាជាងឈើ។ លើសពីនេះ ទៅទៀតនោះ លោកដូណាស់ដូក៏ត្រូវការប្រាក់ផងដែរ។

មានតែលោកដូណាល់ដូ និងក្រុមជំនុំ២នាក់ ព្រមទាំងប៉ាស្ទ័ររបស់ពួកគេចំនួន៤ប៉ុណ្ណោះ ដែល ជាអ្នកសេវេនដេយអាត់វេនទីស្ទ ដែលបានអនុញ្ញាតឱ្យទៅក្នុងពន្ធធនាគានៅក្នុងទីក្រុងអាខាខៃស៍ (Acacias)។ ហេតុដូច្នេះហើយ បានលោកដូណាល់ដូទូរស័ព្ទសូមជំនួយពីមនុស្ស៣នាក់ទៀត។ លោករ៉េនសេស (Ranses) បាននិយាយថា «ទេ ខ្ញុំជាប់រវល់ហើយលោកគ្រូ»។ លោកពេត្រូ (Pedro) ក៏បានប្រកែកថា គាត់មិនអាចទៅបានដែរ។ រីឯប៉ាស្ទ័ររបស់គាត់វិញ បានសុំទោសលោកដូណាល់ដូ ដោយនិយាយថា គាត់នៅក្នុងទីក្រុងនោះឡើយ។

លោកដូណាល់ដូបានអធិស្ឋានទាំងក្តុកក្តួលចិត្ត។ បន្ទាប់មក នៅពេលដែលនាងចេស៊ូស៊ីតា (Jesusita) ដែលជាភរិយារបស់គាត់ បានសួរនាំគាត់ នោះគាត់បានបកស្រាយថា គាត់មិនចង់អាក់ខាន ការជួបជុំជាមួយអ្នកទោសនោះឡើយ ប៉ុន្តែ មានកិច្ចការចាំបាច់ត្រូវធ្វើ។ នាងចេស៊ូស៊ីតាបាននិយាយថា «បងទៅធ្វើកិច្ចការរបស់បងចុះ ព្រះនឹងដោះស្រាយបញ្ហានេះ»។ ក្រោយមក លោកដូណាល់ដូបានសួរសុខទុក្ខអ្នកទោសចំនួន៣៨នាក់ ដែលបានចូលមកស្តាប់គាត់អធិប្បាយ។ គាត់ក៏បានប្រាប់លោកពេត្រូ ដែល ជាសមាជិកម្នាក់ទៀត ឱ្យទៅជាមួយនឹងគាត់ដែរ។ អ្នកទោសម្នាក់បានសួរថា «តើបុរសម្នាក់ទៀត ដែលបាន មកពីលើកមុននោះនៅឯណា?»។ លោកដូណាល់ដូបានតបទៅវិញថា «យើងមិនបានចាត់អ្នកណាម្នាក់ ឱ្យមកនោះឡើយ»។ អ្នកទោសម្នាក់ទៀតបាននិយាយថា «មិនមែនទេ លោកគ្រូពិតជាបានចាត់ឱ្យគេមកមែន ពីព្រោះបុរសម្នា់ក់នោះបានអធិប្បាយឱ្យពួកយើងស្តាប់»។ លោកដូណាល់ដូបានប្រកែកថា «យើងមិន បានចាត់អ្នកណាម្នាក់ឱ្យមកនោះឡើយ» រួចណែនាំថា ប្រហែលជាអ្នកអធិប្បាយម្នាក់នោះ មកពីនិកាយផ្សេង ទៀតហើយ។ អ្នកទោសម្នាក់បាននិយាយថា «ទេ មិនមែនទេលោកគ្រូ ពួកយើងស្គាល់មនុស្សទាំងអស់ ដែល ពន្ធធនាគារបានអនុញ្ញាតឱ្យចូលមកសួរសុខទុក្ខពួកយើង។ បុរសដែលពួកយើងបានឃើញនោះ មិនដែល មកសួរសុខទុក្ខយើងពីមុនមកនោះឡើយ ហើយគាត់ក៏មិនមែនមកពីនិកាយផ្សេងដែរ»។ លោកដូណាល់ដូ បានសួរនាំអំពីការអធិប្បាយរបស់គាត់ ដោយសង្ឃឹមថា នឹងទទួលបានតម្រុយខ្លះពីអត្តសញ្ញាណរបស់បុរសម្នាក់នោះ។ អ្នកទោសបាននិយាយព្រមគ្នាថា បុរសម្នាក់នោះ បាននិយាយអំពីថ្ងៃសប្បាតហ៍ដែលជា ថ្ងៃទី៧។ លោកដូណាល់ដូបាននិយាយទាំងភ្ញាក់ផ្អើលថា «តើគាត់មានរូបសម្បត្តិបែបណាដែរ?»។ អ្នក ទោសបានពិពណ៌នាអំពីកម្ពស់របស់គាត់ ជាមួយការពាក់អាវក្រណាត់ពណ៌សយ៉ាងសមរម្យ។ ពួកគេបាន និយាយថា គាត់ចេះគម្ពីរខ្លាំងគួរសមដែលមើលដូចជាគ្រូម្នាក់អ៊ីចឹង។ លោកពេត្រូបានប៉ះដៃលោកដូណាល់ដូ រួចនិយាយថា «ទេវតារបស់ព្រះប្រាកដជាបានចូលមកអធិប្បាយឱ្យពួកគេស្តាប់ហើយ លោកគ្រូ ពីព្រោះ មានតែការបកស្រាយបែបនេះទេ ទើបអាចទទួលយកបាន»។ បើទោះជាយ៉ាងនេះក្តី ក៏លោកដូណាល់ដូ ហាក់ដូចជាមិនអស់ចិត្តទាល់តែសោះ។ គាត់បានចូលទៅជួបនឹងអ្នកយាមពន្ធនាគារ ដែលបានចុះឈ្មោះ ឱ្យភ្ញៀវទាំងនោះចូលមកក្នុងពន្ធនាគារ។ អ្នកយាមដែលជាមិត្តរបស់លោកដូណាល់ដូ បានពិនិត្យមើល តាមប្រព័ន្ធកុំព្យូទ័រ ដោយងក់ក្បាលរួចនិយាយថា «នៅថ្ងៃនោះ គ្មានអ្នកណាម្នាក់ចូលមកសួរសុខទុក្ខ អ្នកទោសនោះទេ»។ លោកដូណាល់ដូបានលាន់មាត់ទាំងភ្ញាក់ផ្អើលថា «ឥឡូវនេះ ខ្ញុំគ្មានមន្ទិលសង្ស័យ អ្វីទៀតឡើយ ដែលទេវតារបស់ព្រះអម្ចាស់បានចូលមកបង្រៀនព្រះគម្ពីរជំនួសឱ្យខ្ញុំនោះ»។ គាត់បាន ត្រឡប់ ទៅរកអ្នកទោសវិញ រួចប្រាប់ពួកគេថា ពួកគេប្រាកដជាបានឃើញទេវតារបស់ព្រះពិតមែនហើយ។

៦ឆ្នាំក្រោយមក មានអ្នកទោសជិត៣៨នាក់ បានថ្វាយចិត្តរបស់ពួកគេដល់ព្រះយេស៊ូវ តាម

រយៈពិធីបុណ្យជ្រមុជទឹក។ លោកដូណាល់ដូបាននិយាយថា គាត់មិនដែលភ្លេចថ្ងៃនោះ ឡើយ។ គាត់បាននិយាយថា «ទោះបីជារឿងនេះ មិនមែនរឿងគួរឱ្យជឿក៏ដោយ ក៏ខ្ញុំនៅ តែជឿថា ព្រះបានចាត់អ្នកនាំសាររបស់ទ្រង់ពីស្ថានសួគ៌ឱ្យមកសួរសុខទុក្ខអ្នកទោស ទាំងនោះដែរ»។
