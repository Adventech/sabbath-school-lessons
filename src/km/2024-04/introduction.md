### ប្រធានបទ សំខាន់ៗ នៅក្នុង គម្ពីរ យ៉ូហាន

កម្រាលព្រំនៃសាសន៍ពើស៊ីដ៏ស្រស់ស្អាតមួយ បានព្យួរលក់ នៅក្នុងហាងមួយកន្លែង នៅក្នុងទីក្រុងថេរ៉ាន (Tehran) នៃប្រទេសអ៊ីរ៉ាក់។ នៅលើកម្រាលព្រំ មានត្បាញរូបភាពដ៏ស្រស់ស្អាតមួយ នៃព្រៃនៅក្នុងប្រទេស ស្វីស ជាមួយនឹងភ្នំ ទឹកជ្រោះ បឹងបួ និងផ្ទៃមេឃ ដែលមានពេញទៅដោយពពក។

អ្នកណា ដែលបានចូលទៅមើលហាងនោះ អាចចំណាយពេលពិនិត្យមើល ពីភាពផ្ចិតផ្ចង់នៃការត្បាញកម្រាលព្រំនោះ។ ពួកគេក៏អាចពិនិត្យមើលក្រណាត់ និងប្រភេទនៃពណ៌ ដែលបានប្រើ ដើម្បីលាបពណ៌កម្រាល ព្រំនោះ។

ភ្ញៀង ក៏អាចផ្ចង់មើលទៅលើភាពស្រស់ស្អាត នៃកម្រាលនោះដែរ ដែលមាន៖ ព្រិលនៅលើភ្នំ និងព្រៃជ្រៅ ដ៏ខៀវស្រងាត់។ ពណ៌នៃកម្រាលព្រំ បានល្បាយជាមួយនឹងពណ៌ផ្សេងទៀត ដើម្បីធ្វើឱ្យរូបភាព មានសម្រស់ កាន់តែស្អាត។

ត្រីមាសនេះ យើងនឹងសិក្សាពីកិច្ចការ នៃគំនូរមួយទៀត គឺជាកណ្ឌគម្ពីរយ៉ូហាន ដែលបានអាន នៅក្នុង ព្រះគម្ពីរ ដែលជាព្រះបន្ទូលរបស់ព្រះនោះឯង។ ប្រសិនបើយើងចង់យល់ នូវអ្វីដែលព្រះចង់មានបន្ទូលមកកាន់ យើង នោះយើងត្រូវតែសិក្សាព្រះគម្ពីរ ដោយប្រុងប្រយ័ត្ន។ យើងត្រូវតែក្រឡេកមើលខនីមួយៗ ជំពូកនីមួយៗ ផ្នែកផ្សេងៗគ្នានៅក្នុងកណ្ឌគម្ពីរ ហើយនិងសារ ដែលចេញពីព្រះគម្ពីរ ដោយផ្ទាល់។ ព្រះគម្ពីរទាំងមូល មាន ប្រភពចេញមកពីព្រះវិញ្ញាណបរិសុទ្ធ។ ទ្រង់បានបង្ហាញមនុស្ស នូវអ្វីដែលត្រូវសរសេរ។ ដូច្នេះហើយ យើង ត្រូវតែសិក្សាខនីមួយៗ នៅក្នុងព្រះគម្ពីរ ទុកជាសៀវភៅស្រាវជ្រាវ ដើម្បីឱ្យយើងបានដឹង នូវអ្វីដែលព្រះចង់ មានបន្ទូល មកកាន់យើង។

ត្រីមាសនេះ យើងនឹងព្យាយាមយល់សារនៃកណ្ឌគម្ពីរយ៉ូហាន។ កណ្ឌគម្ពីរយ៉ូហាន គឺជាកណ្ឌគម្ពីរមួយ នៅក្នុងកណ្ឌគម្ពីរទាំង៤ នៃគម្ពីរសញ្ញាថ្មី ដែលស្តីអំពីជីវិតរបស់ព្រះយេស៊ូវ។ ដំណឹងល្អទាំង៤ គឺជាកណ្ឌគម្ពីរ ម៉ាថាយ ម៉ាកុស លូកា និងយ៉ូហាន។ កណ្ឌគម្ពីរយ៉ូហាន មានលក្ខណៈខុសពីរឿង នៃដំណឹងល្អទាំង៣។ តើអ្វីដែលធ្វើឱ្យដំណឹងល្អរបស់លោកយ៉ូហាន ខុសពីគេនោះ? យើងបានអានអំពីការសន្ទនា ដោយផ្ទាល់ រវាង ព្រះយេស៊ូវ និងអ្នកដទៃទៀត។ ការសន្ទនាទាំងនេះ ក៏រួមបញ្ចូល នូវការនិយាយជាមួយនឹងលោកណាថានែល លោកនីកូដេម ស្ត្រីនៅត្រង់អណ្តូង បុរសស្វិតដៃជើង នៅត្រង់អាងទឹកបេថែសដា បុរសដែលបានពិការភ្នែកពី កំណើត លោកឡាសារ និងបងប្អូនស្រីរបស់គាត់ លោកពីឡាត់ លោកពេត្រុស និងថូម៉ាស។ រឿងទាំងនេះ ភាគច្រើន មានតែនៅក្នុងកណ្ឌគម្ពីរយ៉ូហានប៉ុណ្ណោះ។

កណ្ឌគម្ពីរយ៉ូហាន គឺជាព្រះបន្ទូលរបស់ព្រះ មកកាន់យើង។ ដ្បិតព្រះបានប្រទានពាក្យពេចន៍នេះ ដល់ លោកយ៉ូហាន ដើម្បីឱ្យគាត់ប្រគល់ឱ្យយើង។ ព្រះគម្ពីរទាំងមូល ដែលបានប្រគល់មកឱ្យយើង មានប្រភព មកពីព្រះ មិនមែនមកពីគំនិតរបស់មនុស្សលោកនោះទេ។ លោកយ៉ូហាន គឺជាអ្នកបម្រើព្រះ ដែលសុខចិត្ត ធ្វើតាមទ្រង់។ ព្រះវិញ្ញាណបរិសុទ្ធ បានប្រទានគោលគំនិត និងប្រធានបទសំខាន់ៗជាច្រើន ដល់លោកយ៉ូហាន ដើម្បីឱ្យគាត់ប្រគល់ឱ្យយើង។ ប្រធានបទទាំងនេះ រួមមាន៖ (១) ព្រះយេស៊ូវ ដែលជាព្រះបន្ទូល (ឡូហ្គូស) ពន្លឺ នំបុ័ង ទឹក ព្រះវិញ្ញាណបរិសុទ្ធ ការរួបរួមគ្នាតែមួយ បាដិហារ្យ ទីបន្ទាល់ និងសារសំខាន់ពីព្រះ មិនមែនមកពីគំនិតរបស់មនុស្សលោកនោះទេ។ លោកយ៉ូហាន គឺជាអ្នកបម្រើព្រះ ដែលសុខចិត្តធ្វើតាមទ្រង់។ ព្រះ វិញ្ញាណបរិសុទ្ធ បានប្រទានគោលគំនិត និងប្រធានបទសំខាន់ៗជាច្រើន ដល់លោកយ៉ូហាន ដើម្បីឱ្យគាត់ ប្រគល់ ឱ្យយើង។ ប្រធានបទទាំងនេះ រួមមាន៖ (១) ព្រះយេស៊ូវ ដែលជាព្រះបន្ទូល (ឡូហ្គូស) ពន្លឺ នំបុ័ង ទឹក ព្រះវិញ្ញាណបរិសុទ្ធ ការរួបរួមគ្នាតែមួយ បាដិហារ្យ ទីបន្ទាល់ និងសារសំខាន់។

នៅពេលដែលយើងសិក្សាព្រះគម្ពីរ យើងបានក្រឡេកមើលអត្ថន័យ នៃពាក្យមួយ ឬខមួយចំនួន។ យើង បានពិនិត្យមើលអត្ថន័យនៃពាក្យនោះ នៅក្នុងវចនានុក្រមនៃព្រះគម្ពីរ។ យើងបានពិនិត្យមើលប្រយោគ។ យើង បានក្រឡេកមើល នូវអ្វីដែលបានកើតឡើង នៅក្នុងប្រវត្តិសាស្ត្រ នៅពេលដែលអ្នកសរសេរ បានសរសេរខ ទាំងនេះ ឡើងមកនោះ។ យើងអាចសិក្សាព្រះគម្ពីរ តាមរបៀបភ្ញៀវ ដែលបានចូលទៅទីកម្រាលព្រំដ៏ស្រស់ ស្អាត នៅក្នុងហាងក្រណាត់យ៉ាងដូច្នេះដែរ។ យើងអាចពិនិត្យមើលសាច់ក្រណាត់ ថ្នេរដេរ ពណ៌ក្រណាត់ ហើយនិងផ្នែកខាងក្រោយ នៃក្រណាត់នោះ។ ផ្នែកទាំងអស់នេះ គឺជាផ្នែកសំខាន់។

ប៉ុន្តែ យើងមិនត្រូវបាត់បង់រូបភាពធំ នៅពេលដែលយើងក្រឡេកមើលផ្នែកតូចៗ នៅក្នុងខនៃព្រះគម្ពីរ នោះទេ។ កម្រាលព្រំនៃសាសន៍ពើស៊ី មានសម្រស់ស្រស់ស្អាតខ្លាំង រហូតដល់យើងអាចមើលឃើញភ្នំនៅលើ កម្រាលព្រំនោះ ដែលធ្វើឱ្យយើងគិតថា យើងនៅក្នុងប្រទេសស្វីសនោះពិតមែន។ ទន្ទឹមគ្នានេះ កណ្ឌគម្ពីរ យ៉ូហាន ក៏អាចធ្វើឱ្យយើងគិតថា យើងកំពុងតែនៅ ជាមួយនឹងព្រះយេស៊ូវ នៅក្នុងអំឡុងសម័យរបស់ទ្រង់ នៅលើផែនដីនេះ។ នៅពេលដែលយើងស្គាល់ព្រះយេស៊ូវ នោះយើងនឹងទទួលបានជីវិតអស់កល្ប។ ផ្នែកតូចៗ នៅក្នុងកណ្ឌគម្ពីរយ៉ូហាន មានភារកិច្ចមួយដ៏សំខាន់។ ពួកគេបានជួយឱ្យយើងមើលឃើញ និងយល់ពីរូបភាព ធំ ឬប្រធានបទសំខាន់ នៅក្នុងកណ្ឌគម្ពីរយ៉ូហាន ដែលរូបភាពធំនោះ គឺជាព្រះយេស៊ូវ ដែលជាព្រះអម្ចាស់ និង ជាព្រះអង្គសង្គ្រោះនោះឯង។
