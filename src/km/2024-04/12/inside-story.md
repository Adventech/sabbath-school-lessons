### លោកឃីងស្តុនបានរកឃើញសេចក្តីសង្ឃឹម

គ្រូបង្រៀនព្រះគម្ពីរម្នាក់បានអញ្ជើញមនុស្សមួយចំនួនចូលទៅក្នុងព្រះវិហារសេវេនដេយអាត់វេនទីស្ទនៃក្រុងបេថែល (Bethel) រដ្ឋអាឡាស្កា ដើម្បីចែកចាយទីបន្ទាល់នៃជីវិតរបស់ពួកគេ ជាមួយនឹង ភ្លៀវមកពីមិសិនអាត់វេនទីស្ទ។ តង្វាយនៅថ្ងៃសប្បាតហ៍ទី១៣មួយចំណែកសម្រាប់ត្រីមាសទី១៣ ឆ្នាំ ២០២៤ នឹងត្រូវយកទៅជួសជុល និងពង្រីកព្រះវិហារ ដើម្បីឱ្យមានទីតាំងសម្រាប់មជ្ឈមណ្ឌលបម្រើសហគមន៍នៅក្នុងទីក្រុងបេថែល។ ខ្ញុំបានជិះយន្តហោះទៅកាន់ទីក្រុងដែលមានចំនួន៦៣០០នាក់ ដើម្បី យករឿងនៃបេសកកម្មមកផ្សព្វផ្សាយគម្រោងនេះ។

ប៉ុន្តែ នៅពេលដែលដល់ម៉ោងសម្ភាសន៍រឿង នោះគ្មានអ្នកណាម្នាក់បានបង្ហាញខ្លួននៅក្នុងព្រះវិហារនោះឡើយ។ នាងចយ អេនដឺសុន (Joy Anderson) ដែលជាគ្រូបង្រៀនព្រះគម្ពីរ បានគ្រវីភីស្សា ដ៏ធំមួយ ដែលនាងបានយកពីការិយាល័យមេធាវីរបស់នាងមកកាន់ព្រះវិហារ។ នាងចយ ដែលមានដើម កំណើតមកពីរដ្ឋអាឡាបាម៉ា (Alabama) និងជាអ្នកដឹកនាំព្រះវិហារជាមួយនឹងគ្រូគម្ពីរម្នាក់ទៀត បាន និយាយថា «ទាំងអស់គ្នាមកញុាំភីហ្សាមក ខ្ញុំសង្ឃឹមថា ភីហ្សានេះនឹងលើកទឹកចិត្តដល់អ្នកដែលបានចូល មកថ្វាយបង្គំ»។ ភីហ្សាបានអស់ប្រហែលជាពាក់កណ្តាល នៅពេលដែលដើរចូលទៅក្នុងព្រះវិហារ។ គាត់ បានបង្ហាញអារម្មណ៍ភ្ញាក់ផ្អើល ដោយសារតែគាត់ជាបុគ្គលតែម្នាក់គត់ ដែលមានវត្តមាននៅទីនោះ។ បន្ទាប់មក គាត់បានយកភីហ្សាមួយចំណិត រួចអង្គុយចុះនិយាយអំពីមូលហេតុដែលនាំឱ្យគាត់មកថ្វាយបង្គំ នៅក្នុងព្រះវិហារអាត់វេនទីស្ទ។ អ្នកថែរក្សាព្រះវិហារដែលមានអាយុ៥៩ឆ្នាំបាននិយាយថា គាត់បាន ព្យាយាមផ្តាច់សុរា និងគ្រឿងញៀននៅក្នុងទីក្រុងហ៊ូផឺបេ (Hooper Bay) ដែលជាទីក្រុង ដែលជាតំបន់ ដីសណ្តយូកុន-គូស្កូវីម (Yukon-Kuskokwim) មានមនុស្សចំនួន១៤០០នាក់ ហើយត្រូវចម្ងាយពេលជិះយន្តហោះតូចប្រហែលជា៩០នាទីពីបេថែល។ គាត់បាននិយាយថា «ខ្ញុំចង់ចេញពីជីវិតដ៏អាប់អួមួយ នេះណាស់ ពីព្រោះខ្ញុំបានប្រាប់ខ្លួនខ្ញុំថា បើសិនជាខ្ញុំមិនបានរៀនពីមេរៀននេះ នោះខ្ញុំក៏មិនបានដឹងពីកំហុសរបស់ខ្ញុំដែរ បើសិនជាខ្ញុំមិនបានរៀនពីកំហុសរបស់ខ្ញុំ នោះអ្នកដទៃនឹងគិតថា ខ្ញុំជាមនុស្សល្ងង់ មិនខាន»។

បន្ទាប់មក លោកឃីងស្តុនបានប្តូរទៅបេថែលវិញ ដោយដើរតាមអ្នកស្រុកយូកុន-គូស្កូវីម ដែល បានចាកចេញពីទីក្រុងតូចៗដោយសង្ឃឹមថា នឹងទទួលបានជីវិតល្អប្រសើរនៅក្នុងទីក្រុងបេថែល ដែល ជាសហគមន៍ធំបំផុតនៅភាគខាងលិចនៃរដ្ឋអាឡាស្កា។ ប្រជាជននៅក្នុងតំបន់ដីសណ្តយូកុន-គូស្កូវីម មានជនជាតិអាឡាស្កាប្រហែលជា៨៥ភាគរយ ដែលជាកុលសម្ព័ន្ធយុពអ៊ីគ (Yup’ik) កុលសម្ព័ន្ធខាប់អ៊ីគ (Cup’ik) និងកុលសម្ព័ន្ធអាថាបាស្កាន (Athabascan)។

ថ្ងៃមួយ លោកស្ទីវ (Steve) ដែលជាអ្នកអាត់វេនទីស្ទ និងជាជនជាតិដីសណ្តយូកុន-គូស្កូរីម បានអញ្ជើញលោកឃីងស្តុនមកកាន់ព្រះវិហារបេថែល។ លោកឃីងស្តុនបាននិយាយថា «គាត់បានសួរនាំ ពីបញ្ហារបស់ខ្ញុំ ដូច្នេះហើយ បានជាខ្ញុំទៅមួយនឹងគាត់ដើម្បីឱ្យបានដឹងថា គាត់ទៅទីណា»។

លោកឃីងស្តុនបានរកឃើញកម្លាំងចិត្ត តាមរយៈការចំណាយពេលជាមួយនឹងអ្នកធ្វើដំណើរ ទៅកាន់ព្រះវិហារ ដែលបានយកឈ្នះលើការពឹងផ្អែកទៅលើសុរា និងគ្រឿងញៀន។ គាត់បាននិយាយថា «អ្នកធ្វើដំណើរទៅកាន់ព្រះវិហារបានជួយឱ្យខ្ញុំរួចផុតពីរបស់ទាំងនេះ នេះហើយគឺជាដំណោះស្រាយដែល ព្រះបានជួយដល់ខ្ញុំ»។ គាត់បាននិយាយថា គាត់សប្បាយចិត្តណាស់ ប៉ុន្តែ ទន្ទឹមនឹងនេះ គាត់ក៏ត្រូវតែ

ស្វែងរកចម្លើយបន្ថែមទៀត។ គាត់បាននិយាយទៀតថា «មនុស្សល្អ (ព្រះយេស៊ូវ) ដែលនៅគង់ឋានលើ បានទតឃើញរឿងគ្រប់យ៉ាង ដូច្នេះហើយ យើងទាំងអស់គ្នានឹងទទួលបានចម្លើយឆ្លើយតប ទៅនឹងសំណួររបស់យើង»។
